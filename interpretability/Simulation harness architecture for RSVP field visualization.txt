import bpy
import numpy as np
from pathlib import Path
import sys, os
sys.path.append(os.path.dirname(__file__))
import common
# ---------- parameters ----------
FIELD_FILE = "sim/fields/t020.npz"
THRESH = 0.1
SCALE = 0.15
# --------------------------------
def main():
    common.reset()
    common.camera()
    common.lights()
data = np.load(Path(FIELD_FILE))
    phi = data["phi"]
    v   = data["v"]
    s   = data["s"]
N = phi.shape[0]
    step = 1.0 / N
mat = bpy.data.materials.new("Phi")
    mat.use_nodes = True
    mat.node_tree.nodes["Principled BSDF"].inputs["Emission Strength"].default_value = 2.0
for i in range(N):
        for j in range(N):
            for k in range(N):
                val = phi[i,j,k]
                if val < THRESH:
                    continue
bpy.ops.mesh.primitive_uv_sphere_add(
                    radius=SCALE * val,
                    location=((i-N/2)*step, (j-N/2)*step, (k-N/2)*step)
                )
                obj = bpy.context.active_object
                obj.data.materials.append(mat)
bpy.context.scene.render.filepath = "out/renders/frame.png"
    bpy.ops.render.render(write_still=True)
if __name__ == "__main__":
    main()

"""
RSVP Core: Scalar-Vector-Entropy field evolution
Fields:
  φ(x,t)  : scalar density / potential
  v(x,t)  : vector flow field
  S(x,t)  : entropy field
This is a minimal, stable, explicit-time-step implementation.
"""
import numpy as np
def step(phi, v, s, dt=0.1, dx=1.0):
    """
    One RSVP evolution step.
Args:
        phi : (N,N,N) scalar field
        v   : (3,N,N,N) vector field
        s   : (N,N,N) entropy field
        dt  : timestep
        dx  : spatial resolution
Returns:
        phi_new, v_new, s_new
    """
# ---------- gradients ----------
    grad_phi = np.gradient(phi, dx)
    grad_s   = np.gradient(s, dx)
# ---------- Laplacian ----------
    lap_phi = (
        np.gradient(grad_phi[0], dx, axis=0) +
        np.gradient(grad_phi[1], dx, axis=1) +
        np.gradient(grad_phi[2], dx, axis=2)
    )
# ---------- scalar field ----------
    # φ_t = D∇²φ − α|∇S|²
    grad_s_sq = (
        grad_s[0]**2 +
        grad_s[1]**2 +
        grad_s[2]**2
    )
phi_new = phi + dt * (
        0.1 * lap_phi       # diffusion / smoothing
        - 0.5 * grad_s_sq   # entropy-gradient suppression
    )
# ---------- vector field ----------
    # v relaxes toward −∇φ
    v_target = np.stack([-0.8 * g for g in grad_phi], axis=0)
v_new = (
        0.95 * v +          # memory / inertia
        0.05 * v_target     # alignment
    )
# ---------- entropy field ----------
    # S_t = β|v| + γφ²
    flow_mag = np.sqrt(
        v_new[0]**2 +
        v_new[1]**2 +
        v_new[2]**2
    )
s_new = s + dt * (
        0.1 * flow_mag +    # dissipation from flow
        0.01 * phi**2       # structure cost
    )
# ---------- bounds (numerical hygiene) ----------
    phi_new = np.clip(phi_new, 0.0, 2.0)
    s_new   = np.clip(s_new,   0.0, 5.0)
return phi_new, v_new, s_new
