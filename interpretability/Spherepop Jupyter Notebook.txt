@dataclass
class Event:
    eid: str
    label: str
    value: Any
    parents: Tuple[str, ...]

@dataclass
class History:
    events: Dict[str, Event] = field(default_factory=dict)

    def add(self, event: Event):
        self.events[event.eid] = event

    def lineage(self, eid: str) -> List[str]:
        seen = set()
        order = []
        def dfs(x):
            if x in seen: return
            seen.add(x)
            for p in self.events[x].parents:
                dfs(p)
            order.append(x)
        dfs(eid)
        return order

@dataclass
class Sphere:
    label: str
    thunk: Callable[['History'], Any]
    _eid: str = None

    def pop(self, history: History):
        if self._eid is not None:
            return history.events[self._eid].value, self._eid
        value, parents = self.thunk(history)
        eid = str(uuid.uuid4())
        history.add(Event(eid, self.label, value, parents))
        self._eid = eid
        return value, eid
"""
)

def rsvp_step(label, center, neighbors, kappa=0.2, gamma=0.05):
    def thunk(h):
        sc, ec = center.pop(h)
        sns = [n.pop(h)[0] for n in neighbors]

        lap_phi = sum(n.phi for n in sns)/len(sns) - sc.phi

        phi = sc.phi + kappa * lap_phi
        vx  = sc.vx + kappa * lap_phi
        vy  = sc.vy + kappa * lap_phi
        S   = sc.S + gamma * abs(lap_phi)

        parents = (ec,) + tuple(n._eid for n in neighbors)
        return RSVPCell(phi, vx, vy, S), parents

    return Sphere(label, thunk)
"""
)
