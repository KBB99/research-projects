I'm trying to get this down to one line of bash: #!/bin/bash
# Loop through all mp3 files in the current directory
for file in *.mp3; do
  # Generate a temporary output filename
  output="temp_$file"
# Apply the audio effects: slower speed and pitch shift
  ffmpeg -i "$file" -filter_complex "asetrate=44100*0.944,atempo=0.95,aresample=44100" -q:a 0 "$output"
# Check if ffmpeg was successful
  if [ $? -eq 0 ]; then
    # Remove the original file and rename the new file to the original
    rm "$file"
    mv "$output" "$file"
  else
    echo "Processing failed for $file"
    rm "$output"  # Clean up the temporary file
  fi
done
