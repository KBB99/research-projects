\section{Infinitesimals}

\subsection{Infinitesimal Object}

\begin{definition}[Infinitesimal]
Let $D$ be the object of first-order infinitesimals:
\[
D = \{ d \in \mathbb{R} \mid d^2 = 0 \}.
\]
\end{definition}

\begin{proposition}
$D$ contains only nilpotent elements and the element $0$.
\end{proposition}

\section{Kock--Lawvere Axiom}

\begin{axiom}[Kock--Lawvere]
For every map $f : D \to \mathbb{R}$, there exists a unique pair $(a,b)$ such that:
\[
f(d) = a + b d \quad \text{for all } d\in D.
\]
\end{axiom}

\begin{corollary}
Derivatives exist for all maps $\mathbb{R}\to\mathbb{R}$ and are defined synthetically.
\end{corollary}

\section{Microlinearity}

\begin{definition}[Microlinear Object]
An object $X$ is microlinear if finite limit cones remain limit cones when tested against infinitesimal extensions $X^D$.
\end{definition}

\begin{proposition}
Smooth manifolds, RSVP state spaces, and Polyxan embedding manifolds are microlinear.
\end{proposition}

\section{Tangent Spaces}

\subsection{Synthetic Tangent Bundle}

\begin{definition}[Tangent Bundle]
For a microlinear object $X$, define:
\[
TX = X^D.
\]
Elements of $TX$ are infinitesimal curves $x : D\to X$.
\end{definition}

\begin{proposition}
$TX$ is a microlinear object and carries a natural vector bundle structure.
\end{proposition}

\section{Vector Fields}

\begin{definition}[Vector Field]
A vector field on $X$ is a map:
\[
V : X \to TX
\]
such that $V(x)(0)=x$.
\end{definition}

\begin{proposition}
The RSVP velocity field $\mathbf{v}$ is a synthetic vector field on $M$.
\end{proposition}

\section{Derivatives}

\subsection{Directional Derivative}

Let $f : X\to \mathbb{R}$ and $V$ a vector field.

\begin{definition}
\[
V(f)(x) = f(V(x)(d)) - f(x)
\]
for $d\in D$ yields the directional derivative.
\end{definition}

\begin{proposition}
This definition agrees with classical differentiation on smooth manifolds.
\end{proposition}

\section{Synthetic Differential Forms}

\begin{definition}[1-Form]
A 1-form on $X$ is a map:
\[
\omega : TX \to \mathbb{R}
\]
linear in infinitesimal directions.
\end{definition}

\begin{definition}[Exterior Derivative]
The synthetic exterior derivative $d\omega$ is the alternating sum of pullbacks along infinitesimal 2-simplices.
\end{definition}

\subsection{Scalar Field}

\Phi : M\to \mathbb{R}

\subsection{Vector Field}

\mathbf{v} : M \to TM

\section{Synthetic Flow Equations}

Let $x : D \to M$ be an infinitesimal curve.

\begin{definition}[RSVP Flow]
The RSVP flow is the first-order system:
\[
x(d) = x(0) + \mathbf{v}(x(0))\, d,
\]
satisfied for all $d\in D$.
\end{definition}

\begin{proposition}
Solutions of the synthetic flow coincide with classical solutions of the RSVP PDE system.
\end{proposition}

\section{Curvature}

\subsection{Connection}

\begin{definition}[Affine Connection]
A connection $\nabla$ on $TM$ is a map:
\[
\nabla : \Gamma(TM) \times \Gamma(TM) \to \Gamma(TM)
\]
satisfying synthetic linearity and Leibniz rules.
\end{definition}

\subsection{Riemann Curvature}

\begin{definition}[Synthetic Curvature]
\[
R(X,Y)Z =
\nabla_X \nabla_Y Z
 - \nabla_Y \nabla_X Z
 - \nabla_{[X,Y]} Z.
\]
\end{definition}

\begin{proposition}
Curvature strata of Chapter~13 arise from synthetic sectional curvature values.
\end{proposition}

\section{Jet Bundles}

\subsection{Jets in SDG}

\begin{definition}[1-Jet]
\[
J^1(X) = X^{D^2}
\]
where $D^2 = \{ (d_1,d_2) \mid d_1^2=d_2^2 = d_1 d_2 = 0 \}$.
\end{definition}

\begin{definition}[k-Jet]
\[
J^k(X) = X^{D^k}
\]
where $D^k$ is the $k$-th order infinitesimal object.
\end{definition}

\subsection{RSVP Jet Fields}

\[
J^k(\Phi),\quad J^k(\mathbf{v}), \quad J^k(S)
\]
synthetically encode the RSVP Lagrangian terms of Chapter~15.

\begin{proposition}
Second-order jets determine curvature corrections and higher-order relaxation terms.
\end{proposition}

\section{Embedding Maps in SDG}

\begin{definition}[Synthetic Embedding]
An embedding $\iota : \mathrm{Inc}(\mathcal{G}) \to M$ is monic and microlinear-preserving.
\end{definition}

\begin{proposition}
Curvature matching of Chapter~31 is equivalent to equality of synthetic 2-jets of $\iota$.
\end{proposition}

\section{Synthetic Harmonicity}

\begin{definition}[Synthetic Harmonic Map]
A map $\iota$ is harmonic if:
\[
\Delta \iota = 0
\]
synthetically, where $\Delta$ is defined via $d\,d^\dagger$ on differential forms.
\end{definition}

\begin{proposition}
Harmonic embeddings coincide with critical points of the RSVP energy functional.
\end{proposition}

\section{Modal Operators in SDG}

\begin{proposition}
The modal operator $\Box_{\mathrm{RSVP}}$ corresponds to sheafification under the infinitesimal neighbourhood topology.
\end{proposition}

\section{Infinitesimal Patching}

\begin{definition}[Infinitesimal Patch]
A patch is infinitesimal if it acts trivially on all curves $D\to X$.
\end{definition}

\begin{proposition}
Infinitesimal patches generate the reconciliation monad via infinitesimal thickening.
\end{proposition}

\section{Torsion and Vorticity}

\begin{definition}[Synthetic Torsion]
\[
T(X,Y) = \nabla_X Y - \nabla_Y X - [X,Y].
\]
\end{definition}

\begin{definition}[Synthetic Vorticity]
\[
\omega = d\mathbf{v}^\flat
\]
as a 2-form in SDG.
\end{definition}

\begin{proposition}
Torsion suppression and vorticity decay (Ch.\ 32) hold synthetically for all flows.
\end{proposition}

\section{Microlinear Differential Invariants}

Invariants are defined as:

\[
I : J^k(X) \to \mathbb{R}
\]
constant on infinitesimal neighbourhoods.

\begin{proposition}
RSVP invariants such as curvature, entropy production, and harmonicity are microlinear differential invariants.
\end{proposition}

\section{Synthetic Collapse}

\begin{proposition}
Finite-time collapse to a semantic galaxy is equivalent to the vanishing of synthetic curvature on all infinitesimal extensions.
\end{proposition}
