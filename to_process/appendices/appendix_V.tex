\addcontentsline{toc}{chapter}{Appendix V: Measurement, Observables, and Semantic Diagnostics}

\section{Overview}

This appendix formalises the notion of observables for systems combining:
\begin{itemize}
\item continuous RSVP field dynamics,
\item discrete Polyxan hypergraph rewriting,
\item curvature- and entropy-constrained embeddings,
\item reconciliation-based semantic normalisation.
\end{itemize}

The goal is to describe measurement operators that extract meaningful quantities from hybrid states without violating modal invariants.

\section{State Space and Measurability}

\subsection{Hybrid State Space}

A complete state is a triple
\[
X = (\Phi, \mathbf{v}, S; \mathcal{G}, \iota),
\]
with each component satisfying its respective structural constraints.

\subsection{Sigma-Algebras}

The measurable structure is generated by:
\begin{align*}
\mathcal{F}_{\mathrm{Field}} &= \sigma(\Phi_i, v_i, S_i), \\
\mathcal{F}_{\mathrm{Graph}} &= \sigma(V(\mathcal{G}), E(\mathcal{G}), \mathrm{tag}), \\
\mathcal{F}_{\mathrm{Embed}} &= \sigma(\iota(c)).
\end{align*}

The hybrid sigma-algebra is the product:
\[
\mathcal{F} = \mathcal{F}_{\mathrm{Field}}
  \otimes \mathcal{F}_{\mathrm{Graph}}
  \otimes \mathcal{F}_{\mathrm{Embed}}.
\]

\section{Observables}

\subsection{Local Observables}

Local observables are operators of the form:
\[
\mathcal{O}(X)(x) : \mathsf{Dom} \to \mathbb{R},
\]
including:
\begin{itemize}
\item scalar curvature density,
\item entropy gradient magnitude,
\item local divergence of the vector field,
\item hypergraph vertex curvature label.
\end{itemize}

\subsection{Global Observables}

Global observables include:
\begin{align*}
\mathcal{E}_{\mathrm{field}}(X) &= \sum_i \left( \|\nabla \Phi_i\|^2 + \|\nabla \times \mathbf{v}_i\|^2 \right), \\
\mathcal{E}_{\mathrm{graph}}(X) &= \sum_{c \in \mathrm{Inc}(\mathcal{G})} \mathcal{H}_0(c), \\
\mathcal{E}_{\mathrm{embed}}(X) &= \int_{\iota(\mathcal{G})} \mathcal{R}(y) \, d\mu(y).
\end{align*}

\subsection{Semantic Observables}

Semantic observables quantify structure at the reconciliation level:
\[
\mathsf{depth}(X), \qquad 
\mathsf{redundancy}(X), \qquad
\mathsf{patch\_entropy}(X).
\]

They extract information about quine structure, patch complexity, and assembly index.

\section{Measurement Operators}

A measurement operator is a map:
\[
\mathsf{M} : X \mapsto \mathsf{M}(X) \in \mathcal{D}(Y),
\]
where $\mathcal{D}(Y)$ is a suitable distribution space.

\begin{itemize}
\item Field operators sample $\Phi,v,S$ using bounded probes.
\item Hypergraph operators count pattern instances.
\item Embedding operators compute distances and curvature values.
\end{itemize}

\subsection{Compatibility Conditions}

Operators must satisfy:
\begin{enumerate}
\item locality (bounded influence),
\item curvature preservation,
\item invariance under normalisation,
\item reconciliation compatibility.
\end{enumerate}

\section{Curvature Diagnostics}

\subsection{Curvature Spectrum}

The curvature spectrum is defined by:
\[
\mathsf{Spec}_{\mathcal{R}}(X) = \{ \lambda_i \}_{i=1}^k,
\]
where $\lambda_i$ are eigenvalues of the discrete Laplace–Beltrami operator over the embedding.

\subsection{Curvature Profiles}

A profile function:
\[
c_X(d) = \frac{1}{|\{y : \mathsf{dist}(y,0)=d\}|}
          \sum_{y : \mathsf{dist}(y,0)=d}
          \mathcal{R}(y)
\]
describes radial curvature behaviour.

\section{Entropy Diagnostics}

Entropy metrics include:
\begin{align*}
\mathcal{S}_{\mathrm{field}} &= \sum_i S_i, \\
\mathcal{S}_{\mathrm{graph}} &= \sum_{c} \mathrm{tag}(c), \\
\mathcal{S}_{\mathrm{patch}} &= -\sum_k p_k \log p_k,
\end{align*}
where $p_k$ are normalised frequencies of atomic patches.

\section{Semantic Diagnostics}

\subsection{Assembly Index}

The assembly index is computed as the minimal number of quine extensions required to assemble a structure.

\subsection{Kullback–Leibler Divergence}

For two semantic manifolds $M_x$ and $M_y$ with distributions $P_x$ and $P_y$:
\[
D_{\mathrm{KL}}(P_x \| P_y)
\]
is defined when both distributions are curvature-compatible.

\subsection{Harmonicity Score}

A harmonicity score is determined by:
\[
H(X) = \int_{\iota(\mathcal{G})} \|\Delta \Phi\|^2 + \|\nabla \cdot \mathbf{v}\|^2 \, d\mu.
\]

\section{Diagnostic Pipelines}

Diagnostic pipelines combine measurements into analysis routines:
\begin{enumerate}
\item field analysis,
\item graph rewriting analysis,
\item curvature comparison,
\item semantic normalisation audit,
\item reconciliation residual assessment.
\end{enumerate}

Each pipeline is defined as a sequence of measurement operators satisfying invariance conditions.

\section{Comparison Metrics}

Hybrid distance metrics include:
\[
d(X,Y) = \alpha\, d_{\mathrm{field}} + 
         \beta\, d_{\mathrm{graph}} +
         \gamma\, d_{\mathrm{embed}} +
         \delta\, d_{\mathrm{semantic}},
\]
with weights tuned to preserve modal invariants.

\section{Convergence Diagnostics}

Convergence is assessed by:
\begin{itemize}
\item stabilisation of curvature spectrum,
\item vanishing of harmonicity score,
\item stabilisation of assembly index,
\item reduction of KL divergence across iterations.
\end{itemize}

Convergence implies that the system has reached a normalised, curvature-minimising fixed point.
