\section{Statistical Manifolds}

\subsection{Smooth Families of Distributions}

A statistical manifold is a smooth manifold $M$ equipped with a family of probability densities
\[
p(x \mid \theta), \qquad \theta \in M,
\]
with $p(\cdot \mid \theta)$ smooth in $\theta$.

\begin{definition}[Score Function]
The score at $\theta$ is
\[
u_i(x,\theta) = \frac{\partial}{\partial \theta^i} \log p(x\mid\theta).
\]
\end{definition}

\section{Fisher Information Metric}

\subsection{Definition}

The Fisher metric on $M$ is
\[
g_{ij}(\theta)
  = \int u_i(x,\theta)u_j(x,\theta)p(x\mid\theta)\,dx.
\]

\subsection{Properties}

\begin{itemize}
\item $g_{ij}$ is positive definite.
\item It transforms as a Riemannian metric under reparametrisation.
\item It is invariant under sufficient statistics.
\end{itemize}

\section{Affine Connections in Information Geometry}

\subsection{$\alpha$-Connections}

For $\alpha \in \mathbb{R}$, the $\alpha$-connection is defined by
\[
\Gamma_{ijk}^{(\alpha)}
  = \int \frac{\partial^2}{\partial\theta^i\partial\theta^j}\log p(x\mid\theta)\,
    u_k(x,\theta)\,p(x\mid\theta)\,dx
  + \frac{1-\alpha}{2} T_{ijk},
\]
where $T_{ijk}$ is the skewness tensor.

\subsection{Dual Connections}

The $\alpha$ and $(-\alpha)$ connections are dual in the sense:
\[
X\,g(Y,Z) = g(\nabla_X^{(\alpha)} Y, Z) + g(Y, \nabla_X^{(-\alpha)} Z).
\]

\section{Divergence Functions}

\subsection{Csiszár $f$-Divergences}

Given strictly convex $f$ with $f(1)=0$, the $f$-divergence is
\[
D_f(p\|q)
  = \int q(x) f\!\left(\frac{p(x)}{q(x)}\right) dx.
\]

\subsection{Bregman Divergences}

For a convex function $F$,
\[
D_F(\theta\|\eta)
  = F(\theta) - F(\eta)
    - \langle \nabla F(\eta), \theta - \eta \rangle.
\]

\section{Kullback–Leibler Divergence}

\subsection{Definition}

\[
D_{\mathrm{KL}}(p\|q)
  = \int p(x)\log\frac{p(x)}{q(x)} dx.
\]

\subsection{Convexity and Positivity}

\begin{itemize}
\item $D_{\mathrm{KL}}(p\|q)\ge 0$ with equality iff $p=q$ a.e.
\item It is convex in the pair $(p,q)$.
\item It is a Bregman divergence for the negative Shannon entropy.
\end{itemize}

\section{Geodesics and Statistical Curvature}

\subsection{Exponential and Mixture Geodesics}

The exponential geodesic between distributions $p_0, p_1$ is
\[
p_t(x)
  = \frac{p_0(x)^{1-t} p_1(x)^{t}}{Z(t)}.
\]

The mixture geodesic is
\[
p_t(x)
  = (1-t)p_0(x) + tp_1(x).
\]

\subsection{Statistical Curvature}

The curvature of the $\alpha$-connection quantifies deviation from an exponential family structure.

For a statistical curve $\theta(t)$,
\[
\kappa(t)
  = \| \nabla_{\dot{\theta}}^{(\alpha)} \dot{\theta} \|_{g}.
\]

\section{Information-Geometric Projection}

\subsection{Orthogonality}

Given a submanifold $N \subseteq M$, the projection of $\theta$ onto $N$ in the Fisher metric minimises
\[
D_{\mathrm{KL}}(\theta\|\eta), \qquad \eta \in N.
\]

\subsection{Pythagorean Theorems}

For an exponential family $E$ and mixture family $M$,
\[
D_{\mathrm{KL}}(p\|r)
  = D_{\mathrm{KL}}(p\|q) + D_{\mathrm{KL}}(q\|r),
\]
whenever $q = P_E(p)$ or $q = P_M(p)$.

\section{Dual Flatness}

\subsection{Definition}

A statistical manifold is dually flat if:

\begin{itemize}
\item the $\alpha=1$ and $\alpha=-1$ connections are flat,
\item the metric arises from the Hessian of a convex potential.
\end{itemize}

\subsection{Coordinates}

Dual coordinates $(\theta,\eta)$ satisfy:
\[
\eta_i = \frac{\partial F}{\partial \theta^i}, \qquad
g_{ij} = \frac{\partial^2 F}{\partial\theta^i\partial\theta^j}.
\]

\section{Information Flow and Gradient Dynamics}

\subsection{Natural Gradient}

For objective $L(\theta)$,
\[
\dot{\theta}
  = -g^{-1}(\theta)\nabla_\theta L(\theta)
\]
is the steepest descent in distribution space.

\subsection{KL Gradient Flow}

KL gradient flow on the space of densities satisfies
\[
\partial_t p_t
  = \nabla \cdot \!\left( p_t \nabla \frac{\delta L}{\delta p} \right).
\]

\section{Information Geometry of Curvature Fields}

\subsection{Curvature Embedding}

Given scalar curvature field $\mathcal{R}$ and entropy field $S$,
associate distributions
\[
p_{\mathcal{R}}(x) = \frac{e^{-\beta \mathcal{R}(x)}}{Z_{\mathcal{R}}},
\qquad
p_{S}(x) = \frac{e^{-\beta S(x)}}{Z_{S}},
\]
with Fisher metrics induced from their gradients.

\subsection{Curvature Divergence}

Define curvature divergence by:
\[
D_{\mathrm{curv}}(p_{\mathcal{R}_1}\|p_{\mathcal{R}_2})
  = D_{\mathrm{KL}}(p_{\mathcal{R}_1}\|p_{\mathcal{R}_2}).
\]

\section{Divergence Minimisation and Alignment}

\subsection{Optimal Alignment}

Given semantic manifolds $(\mathcal{M}_1,p_1)$ and $(\mathcal{M}_2,p_2)$, the divergence-minimising alignment problem is:
\[
\inf_{\pi\in\Pi(p_1,p_2)}
  \mathbb{E}_{\pi}\!\left[
      |\mathcal{R}_1(x)-\mathcal{R}_2(y)|
      +
      |S_1(x)-S_2(y)|
    \right].
\]

\subsection{Zero-Divergence Condition}

Zero divergence occurs iff curvature and entropy fields match almost everywhere.
