\section{Measurable Structures on Semantic Manifolds}

\subsection{Borel $\sigma$-Algebras}

Let $(\mathcal{M},g)$ be a smooth manifold equipped with the RSVP metric $g$.
The Borel $\sigma$-algebra $\mathcal{B}(\mathcal{M})$ is generated by all open sets of $\mathcal{M}$.

\begin{definition}[Semantic Borel Space]
A semantic Borel space is a pair $(\mathcal{M},\mathcal{B}(\mathcal{M}))$ where $\mathcal{M}$ is a semantic manifold arising from a stratified embedding of a Polyxan hypergraph.
\end{definition}

\section{Radon Measures and Regularity}

\subsection{Radon Measures}

A measure $\mu$ on $(\mathcal{M},\mathcal{B}(\mathcal{M}))$ is a Radon measure if:
\begin{itemize}
\item $\mu$ is locally finite,
\item $\mu$ is inner regular (from compact sets),
\item $\mu$ is outer regular (from open sets).
\end{itemize}

\begin{definition}[Semantic Radon Measure]
A semantic Radon measure is a Radon measure absolutely continuous with respect to the Riemannian volume form $dV_g$.
\end{definition}

\section{Probability Measures on Field Configurations}

\subsection{Gibbs Measures}

Given a scalar field $\Phi$ and entropy field $S$, define the Gibbs density
\[
p(x) = \frac{1}{Z} \exp\bigl(-\beta \mathcal{E}[\Phi,S](x)\bigr),
\]
with partition function
\[
Z = \int_{\mathcal{M}} \exp\bigl(-\beta \mathcal{E}[\Phi,S](x)\bigr) dV_g(x).
\]

\begin{definition}[Semantic Probability Measure]
A probability measure $\mathbb{P}$ on $\mathcal{M}$ is semantic if $\mathbb{P}(dx) = p(x)\,dV_g(x)$ for some Gibbs density $p$.
\end{definition}

\section{Integration on Stratified Spaces}

\subsection{Stratification}

Let
\[
\mathcal{M} = \bigsqcup_{k=0}^{d} \mathcal{M}^{(k)},
\]
where $\mathcal{M}^{(k)}$ is the $k$-codimensional stratum of the semantic manifold.

\subsection{Piecewise Integration}

For measurable $f:\mathcal{M}\to\mathbb{R}$,
\[
\int_{\mathcal{M}} f(x)\, dV_g(x)
  = \sum_{k=0}^{d} \int_{\mathcal{M}^{(k)}} f(x)\, dV_{g^{(k)}}(x).
\]

The restriction $dV_{g^{(k)}}$ is the induced volume measure on the stratum.

\section{Pushforward and Pullback of Measures}

\subsection{Pushforward}

For measurable $\iota:\mathcal{G}\to\mathcal{M}$,
\[
\iota_\# \mu(A) = \mu(\iota^{-1}(A)).
\]

\subsection{Pullback}

If $\nu$ is absolutely continuous with respect to $dV_g$, the pullback measure on $\mathcal{G}$ is
\[
\iota^\# \nu(B) = \nu(\iota(B)).
\]

\section{Couplings and Transport}

\subsection{Optimal Couplings}

Given $(\mathcal{M}_1,p_1)$ and $(\mathcal{M}_2,p_2)$, the set of couplings is
\[
\Pi(p_1,p_2)
  = \bigl\{ \pi : \text{$\pi$ has marginals $p_1$ and $p_2$} \bigr\}.
\]

\subsection{Transport Cost}

Given cost $c:\mathcal{M}_1\times\mathcal{M}_2\to\mathbb{R}$,
\[
\mathcal{T}_c(p_1,p_2)
  = \inf_{\pi\in\Pi(p_1,p_2)} \int c(x,y)\, d\pi(x,y).
\]

\section{Absolute Continuity and Divergence}

\subsection{Absolute Continuity}

$p_1 \ll p_2$ if $p_1(A)=0$ whenever $p_2(A)=0$.

\subsection{Radon–Nikodym Derivative}

If $p_1\ll p_2$, then
\[
\frac{dp_1}{dp_2}(x)
\]
exists almost everywhere.

\section{KL Divergence and Curvature Mismatch}

\subsection{KL Divergence}

\[
D_{\mathrm{KL}}(p_1\|p_2)
  = \int p_1(x) \log \frac{p_1(x)}{p_2(x)} dV_g(x).
\]

\subsection{Curvature-Adjusted Divergence}

Given scalar curvature $\mathcal{R}_i$ and entropy $S_i$,
\[
D_{\mathrm{curv}}(p_1\|p_2)
  = \int p_1(x)
      \Bigl(
        |\mathcal{R}_1 - \mathcal{R}_2| +
        |S_1 - S_2|
      \Bigr)
      dV_g(x).
\]

\section{Expectation and Variance of Field Quantities}

\subsection{Expectation}

For measurable $F:\mathcal{M}\to\mathbb{R}$,
\[
\mathbb{E}[F] = \int_{\mathcal{M}} F(x) p(x)\, dV_g(x).
\]

\subsection{Variance}

\[
\mathrm{Var}[F]
  = \mathbb{E}[F^2] - \mathbb{E}[F]^2.
\]

\section{Convergence of Measures}

\subsection{Weak Convergence}

$p_n \Rightarrow p$ if
\[
\int f\, dp_n \to \int f\, dp
\]
for every bounded continuous $f$.

\subsection{Convergence in KL}

\[
D_{\mathrm{KL}}(p_n\|p) \to 0
  \;\Longleftrightarrow\;
p_n \to p \text{ in total variation}.
\]

\section{Measure-Theoretic Stability Under Reconciliation}

\subsection{Stability}

For reconciliation map $\mathcal{R}$,
\[
(\mathcal{R}\circ\iota)_\#\mu
  \;\text{is absolutely continuous with respect to}\;
\iota_\#\mu.
\]

\subsection{Entropy Monotonicity}

If $\iota'$ is the reconciled embedding,
\[
\int S(\iota'(x))\, d\mu(x)
  \le
  \int S(\iota(x))\, d\mu(x).
\]

\section{Measure-Theoretic Characterisation of the Universal Quine}

\subsection{Canonical Measure}

The universal media quine $\mathfrak{U}$ induces the canonical measure
\[
p_{\mathfrak{U}}(x) = \frac{1}{Z_\mathfrak{U}}\, e^{-\beta \mathcal{R}_\mathfrak{U}(x)}.
\]

\subsection{Uniqueness}

If $p$ is any semantic Gibbs measure with zero curvature mismatch relative to $\mathfrak{U}$, then
\[
p = p_{\mathfrak{U}}
\quad\text{a.e.}
\]
