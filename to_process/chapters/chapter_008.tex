\chapter{Field--Graph Coupling via Semantic Embeddings}
\label{ch:coupling}

\section{Overview}

The preceding chapters introduced three major structures:

\begin{enumerate}
  \item the semantic manifold $\mathcal{M}$ with RSVP fields $(\Phi,\mathbf{v},S)$;
  \item the semantic sheaf and cognitive rewrite system;
  \item and the geometry and energetics of embedded Polyxan hypergraphs.
\end{enumerate}

This chapter establishes the \emph{unified dynamical system} obtained when these
components are coupled through a sheaf-theoretic embedding functor:

\[
\mathsf{E} : 
\mathsf{Hyp}_{\mathcal{T}}
\longrightarrow 
\mathsf{Sh}(\mathcal{M}),
\]

where $\mathsf{Hyp}_{\mathcal{T}}$ is the typed hypergraph category (from
Ch.~\ref{ch:typed-hypergraphs}) and $\mathsf{Sh}(\mathcal{M})$ is the sheaf category introduced in Chapter~\ref{ch:semantic-manifold}.

Coupling occurs through three layers:

\begin{itemize}
  \item \textbf{geometric coupling:} embeddings into $\mathcal{M}$ generate curvature and receive metric feedback,
  \item \textbf{energetic coupling:} RSVP fields determine hypergraph tension and diffusion,
  \item \textbf{cognitive coupling:} rewrite operations produce structural changes that backreact on fields.
\end{itemize}

The result is a closed semantic–geometric–computational dynamical system.


\section{The Semantic Embedding Functor}

Let $H$ be a typed hypergraph.  
An embedding of $H$ into $\mathcal{M}$ consists of:

\[
\iota_H : H \longrightarrow \mathcal{M},
\qquad 
v \mapsto \iota_H(v),
\qquad
e \mapsto \iota_H(e).
\]

\subsection{Functoriality}

Define the embedding functor:

\[
\mathsf{E}(H)
:=
\left\{
  U \subseteq \mathcal{M}
  \mapsto
  \mathrm{Hom}_{\mathsf{Hyp}_{\mathcal{T}}}
    (H|_U,\, 
     \mathscr{P}(U))
\right\},
\]

where:

\begin{itemize}
  \item $H|_U$ is the restriction of $H$ to nodes whose images intersect $U$,
  \item $\mathscr{P}$ is the Polyxan atom sheaf.
\end{itemize}

This construction yields a sheaf of hypergraph instances on $\mathcal{M}$.


\subsection{Compatibility with RSVP Fields}

The embedding is \emph{metric-compatible}:

\[
\iota_H^\ast g_{\mu\nu}
=
g^{(H)}_{\mu\nu}
\]

and \emph{entropy-compatible}:

\[
S(\iota_H(v)) \ge S(\iota_H(e)) \quad\text{for all } v\in e.
\]

This ensures that embeddings respect semantic granularity.


\section{Coupling Layer I: Geometric Feedback}

\subsection{Pushforward curvature}

Embedding induces curvature:

\[
\mathcal{K}_H(x)
:=
\sum_{v\in V(H)} 
\mathrm{Ric}\!\left(\iota_H(v)\right)
\delta(x-\iota_H(v)).
\]

\subsection{Pullback curvature}

Conversely, hypergraph tension modifies curvature through:

\[
\Xi_{\mu\nu}
=
\frac{\delta E_H}{\delta g^{\mu\nu}},
\]

as defined in Chapter~\ref{ch:curvature-hypergraph}.

Thus the curvature equation becomes:

\[
\mathrm{Ric}_{\mu\nu}
=
T_{\mu\nu}
+
\Xi_{\mu\nu}
-
\frac{1}{2}g_{\mu\nu}(R+\mathrm{tr}\,\Xi).
\]


\section{Coupling Layer II: Energetic Interaction}

\subsection{Hypergraph tension as a field source}

Node embeddings obey:

\[
\partial_t \iota_H(v)
=
-\,\operatorname{grad}_g E_H
+
\mathbf{v}(\iota_H(v)),
\]

so RSVP flow $\mathbf{v}$ advects hypergraphs while curvature gradients reshape them.

\subsection{Field equations with hypergraph sources}

Variation of the RSVP action with respect to $\Phi$ yields:

\[
\nabla^2 \Phi 
=
\frac{\partial V}{\partial \Phi}
+
\gamma \nabla\cdot \mathbf{v}
+
J_\Phi^{(H)},
\]

where the hypergraph source is:

\[
J_\Phi^{(H)}
=
\sum_{v\in V(H)}
m(v)\, \delta(x-\iota_H(v)).
\]

Entropy and flow fields obtain analogous source terms:

\begin{align*}
\beta\nabla_\mu(S^{-1}\nabla^\mu S) &= \frac{\partial V}{\partial S} + J_S^{(H)}, \\
\gamma \nabla_\mu \Phi &= J_{\mathbf{v}}^{(H)}.
\end{align*}

These encode:

\begin{itemize}
  \item semantic inertia,
  \item cognitive salience,
  \item and rewiring demands.
\end{itemize}


\section{Coupling Layer III: Cognitive Rewrite Backreaction}

The rewrite dynamics (Ch.~\ref{ch:lsystem}) induce changes in $H$.

\subsection{Rewrite-induced hypergraph current}

A rewrite rule
\[
\rho : \mathscr{P} \to \mathscr{P}^{\otimes k}
\]
induces a hypergraph current:

\[
J^{(\rho)}_{H}
=
\frac{\partial H}{\partial t}
=
\rho(H) - H.
\]

\subsection{Field response to rewrites}

The RSVP fields respond as:

\[
\partial_t \Phi
=
-\frac{\delta\mathcal{S}}{\delta \Phi}
+
\alpha_\Phi\, \mathrm{div}\,J_H^{(\rho)},
\]

\[
\partial_t S
=
-\frac{\delta\mathcal{S}}{\delta S}
+
\alpha_S\, \mathrm{ev}(J_H^{(\rho)}),
\]

\[
\partial_t \mathbf{v}
=
-\frac{\delta\mathcal{S}}{\delta \mathbf{v}}
+
\alpha_v\, \nabla(\mathrm{mass}(J_H^{(\rho)})).
\]

Here $\alpha_\Phi,\alpha_S,\alpha_v$ are coupling constants.

These describe cognitive activity feeding back into the physical smoothing process.


\section{Functorial Compatibility and Coherence}

\subsection{Cartesian naturality}

The core compatibility requirement is that the embedding functor commutes with rewriting:

\[
\mathsf{E}\circ\rho
=
(\mathsf{U}\circ\mathsf{E}),
\]

where $\mathsf{U}$ is the L-system unfolding functor from Chapter~\ref{ch:lsystem}.

This ensures:

\[
\text{rewrite first, embed later}
\quad=\quad
\text{embed first, rewrite in place}.
\]

Thus the semantic manifold implements rewrite dynamics as geometric flows.


\subsection{Monoidal compatibility}

For parallel hyperedge production:

\[
\mathsf{E}(H_1 \otimes H_2)
=
\mathsf{E}(H_1)\otimes\mathsf{E}(H_2).
\]

This ensures that graph parallelism matches sheaf parallelism.


\section{Global Dynamics}

Combine geometric, energetic, and rewrite couplings:

\[
\boxed{
\begin{aligned}
\partial_t g_{\mu\nu}
&=
-2(T_{\mu\nu}+\Xi_{\mu\nu}),\\[4pt]
\partial_t \Phi
&=
\nabla^2 \Phi 
+ \gamma\nabla\cdot\mathbf{v}
+ J_\Phi^{(H)}
+ \alpha_\Phi\mathrm{div}\,J^{(\rho)}_{H},\\[4pt]
\partial_t S
&=
\beta\nabla_\mu(S^{-1}\nabla^\mu S)
+ J_{S}^{(H)}
+ \alpha_S \mathrm{ev}(J^{(\rho)}_{H}),\\[4pt]
\partial_t \mathbf{v}
&=
\gamma^{-1} J_{\mathbf{v}}^{(H)}
+ \alpha_v\nabla(\mathrm{mass}(J^{(\rho)}_{H})),\\[4pt]
\partial_t H
&=
-\operatorname{grad}_g E_H
+ \mathbf{v}\cdot\nabla H
+ J^{(\rho)}_{H}.
\end{aligned}
}
\]

This system couples:

\begin{itemize}
  \item geometry (metric flow),
  \item entropy (semantic density),
  \item affect (entropy-weighted rewrites),
  \item cognition (rewrite rules),
  \item and structure (hypergraph embeddings).
\end{itemize}


\section{Conservation and Invariants}

By diffeomorphism invariance of the total action,
\[
\nabla^\mu(T_{\mu\nu}+\Xi_{\mu\nu})=0.
\]

There exists a conserved \textbf{semantic charge}:

\[
Q
=
\int_{\Sigma}
\left(S + \mathrm{mass}(H)\right)
\sqrt{h}\, d^3x
\]

for any spatial hypersurface $\Sigma$.

Rewrite operations preserve $Q$ up to entropy dissipation.
