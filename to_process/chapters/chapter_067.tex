\section{The EHR System in Dependent Type Theory}

A Polyxan hypergraph is encoded in $\mathbb{T}_{\mathrm{Polyx}}$ as the inductive type
\[
\mathsf{Polyx} := \mu X.\;
  \mathsf{Cell}(X)
  \,+\, \mathsf{Edge}(X \times X)
  \,+\, \mathsf{Tag}(X \times \mathsf{Type}),
\]
with stratified incidence automatically generated by the inductive structure.

\begin{definition}[EHR Rewrite Judgment]
An \emph{Elementary--Hoist--Reduce} rewrite step is represented by the type family
\[
\mathsf{EHR}(G,G')
  := \Sigma\,r:\mathsf{Rule},\;
     \mathsf{Applies}(r,G)\;\times\; (G' = \mathsf{apply}(r,G)).
\]
\end{definition}

The three primitive rule families are:

\begin{itemize}
\item $\mathsf{E}$: atomic extraction rules,
\item $\mathsf{H}$: hoisting rules for lifting stratified structure,
\item $\mathsf{R}$: reduction rules certified as curvature-decreasing.
\end{itemize}

All rules are indexed by dependent types ensuring that the output hypergraph is 
well-typed, properly stratified, and curvature-monotone.

\section{Modal Soundness of EHR Rewriting}

\begin{theorem}[Modal Soundness]
\label{thm:67-sound}
For any Polyxan hypergraphs $G$ and $G'$, if
\[
\mathsf{EHR}(G,G')
\]
holds, then the modal predicate $\Box_{\mathrm{Polyx}}$ is preserved:
\[
\Box_{\mathrm{Polyx}}(P(G)) \;\Longrightarrow\; \Box_{\mathrm{Polyx}}(P(G')).
\]
\end{theorem}

\begin{proof}
Fix a predicate $P$ satisfying the modal guard for Polyxan rewriting 
(Chapter~61):
\[
\forall H,H'.\; \mathsf{EHR}(H,H') \to P(H) \to P(H').
\]
Suppose $\mathsf{EHR}(G,G')$ is witnessed by a rule $r$ and an equality proof 
$G'=\mathsf{apply}(r,G)$.  
By the definition of $\Box_{\mathrm{Polyx}}$, the assumption 
$\Box_{\mathrm{Polyx}}(P(G))$ states that $P$ holds for $G$ under any 
rewrite sequence.  
Since $G \to G'$ is one such step, the modal guard yields $P(G')$.  
Thus $P$ is preserved, completing the proof.
\end{proof}

\section{Termination: Verified Normalisation to Quine Form}

Termination is the essential property that makes Polyxan rewriting 
a verification-friendly system.

\begin{theorem}[Verified Termination]
\label{thm:67-termination}
Every sequence of EHR rewrites terminates in a finite number of steps at the 
unique quine-stable normal form $\mathfrak{U}$.
\end{theorem}

\begin{proof}
Each reduction rule in the $\mathsf{R}$ family is indexed by a dependent type 
certifying curvature decrease:
\[
\mathcal{H}_0[G'] < \mathcal{H}_0[G].
\]
Since $\mathcal{H}_0$ is a natural-number valued stratified curvature measure 
(Ch.~25), the descent is strictly monotone.  
Extraction and hoist rules either preserve or reduce curvature.  
Thus any infinite sequence would generate an infinite strictly descending chain 
in $\mathbb{N}$, which is impossible.  
Uniqueness of the final normal form follows from the confluence theorem proved 
in Chapter~25.
\end{proof}

\section{Completeness of Modal Interpretation}

\begin{theorem}[Modal Completeness]
\label{thm:67-complete}
Let $P$ be any predicate on Polyxan hypergraphs.  
Then
\[
\Box_{\mathrm{Polyx}}(P(G))
\]
holds if and only if $P$ is preserved by every EHR rewrite step.
\end{theorem}

\begin{proof}
The forward direction is immediate from the definition of $\Box_{\mathrm{Polyx}}$.  
For the converse, consider an arbitrary rewrite sequence
\[
G = G_0 \to G_1 \to \cdots \to G_n.
\]
If $P$ is preserved by every rewrite step, induction on $n$ gives $P(G_n)$.  
Since every such $G_n$ occurs in some EHR path, the modal necessity predicate 
holds exactly when $P$ is rewrite-invariant.  
This is the Kripke-style completeness condition specialised to Polyxan 
rewriting.
\end{proof}

\section{Synthetic Duality and EHR Rewriting}

The duality functor $\mathbb{R}$ constructed in Chapter~62 embeds the EHR 
system into the RSVP flow.  
Thus discrete rewriting corresponds to continuous curvature descent.

\begin{theorem}[Dual Interpretation]
\[
\mathbb{R}(G \to G') \;=\; 
\text{a curvature-decreasing RSVP flow segment}.
\]
\end{theorem}

\begin{proof}
Each reduction rule is certified by a curvature-decrease witness in 
$\mathbb{T}_{\mathrm{RSVP}}$.  
Extraction and hoist correspond to stratified harmonic extensions in the RSVP 
manifold.  
Since $\mathbb{R}$ preserves and reflects both modal operators (Ch.~62), the 
image of any EHR path is a curvature descent path.
\end{proof}

\section{The Verified EHR Kernel}

\begin{definition}[Verified EHR Kernel]
A rewriting kernel is \emph{verified} if it is a recursive function
\[
\mathsf{ehr\_step} : \mathsf{Polyx} \to \mathsf{Polyx}
\]
that:
\begin{enumerate}
\item preserves modal correctness ($\Box_{\mathrm{Polyx}}$),
\item strictly decreases $\mathcal{H}_0$ on non-normal inputs,
\item returns $\mathfrak{U}$ on all quine-stable inputs.
\end{enumerate}
\end{definition}

\begin{theorem}[Correctness of the Verified Kernel]
For every Polyxan hypergraph $G$,  
\[
\mathsf{ehr\_step}(G)
\]
is the unique next state in the verified normalisation sequence, and repeated 
iteration reaches $\mathfrak{U}$ in finitely many steps.
\end{theorem}

\begin{proof}
The kernel uses rule-selection indexed by modal guards, ensuring soundness; 
the curvature descent certificate ensures termination; and uniqueness follows 
from confluent normalisation.
\end{proof}
