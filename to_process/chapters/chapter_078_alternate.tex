\section{Distributed Configuration Space}

Let $\mathsf{Config}^{\mathcal{N}}$ denote the configuration space of
all possible distributed states over a network~$\mathcal{N}$:

\[
\mathsf{Config}^{\mathcal{N}}
  :=
  \prod_{\alpha \in \mathcal{N}}
    \Bigl(
      \Phi_\alpha, \mathbf{v}_\alpha, S_\alpha,
      \iota_\alpha, \mathcal{G}_\alpha,
      \mathsf{Queue}_\alpha
    \Bigr),
\]

equipped with the product topology and the partial-order structure
generated by:

\begin{itemize}
\item flow steps on each patch,
\item rewrite batches,
\item halo exchanges,
\item and node-to-node drift reconciliation.
\end{itemize}

Define the global asynchronous update operator:

\[
\mathcal{A} : \mathsf{Config}^{\mathcal{N}}
  \longrightarrow \mathsf{Config}^{\mathcal{N}},
\]

where $\mathcal{A}$ applies \emph{any} set of:
flow updates, rewrite reductions, halo exchanges, or message receipts,
in \emph{any} order, on \emph{any} subset of nodes.


\section{Idempotence of Semantic Collapse}

\begin{theorem}[Asynchronous Idempotence]
\label{thm:async-idempotence}
For every network~$\mathcal{N}$ and every configuration
$C \in \mathsf{Config}^{\mathcal{N}}$,
the limit
\[
\lim_{k\to\infty} \mathcal{A}^k(C)
\]
exists and is independent of update order, scheduler interleaving,
or communication timing.  
Let this limit be written $\mathbf{Collapse}(C)$.

Then:
\[
\mathbf{Collapse}(\mathbf{Collapse}(C)) \;=\; \mathbf{Collapse}(C).
\]
\end{theorem}

\begin{proof}
Existence and uniqueness of the limit follow from:
\begin{enumerate}
\item monotone strict decrease of the total energy
      $\mathcal{E}_{\mathrm{total}}$ except at critical points
      (Ch.~32),
\item strict decrease of Polyxan curvature under rewrites
      (Ch.~25, Ch.~33),
\item and the drift-compositional determinism of parallel chains
      (Ch.~76).
\end{enumerate}

Since the limit is a critical point of the coupling functional,
it is a semantic galaxy (Ch.~34).

Semantic galaxies are fixed points of all updates:
\[
\mathcal{A}(G) = G.
\]

Thus applying $\mathbf{Collapse}$ again yields the same object.
\end{proof}

Ch.~35--36 show that all galaxies collapse through the universal critical point
and relax uniquely to~$\mathfrak{U}$.
\end{proof}

\section{Collapse as a Distributed Projection}

\begin{definition}[Global Semantic Projection]
Define the \emph{semantic projection}
\[
\mathbf{Collapse} :
  \mathsf{Config}^{\mathcal{N}}
  \longrightarrow
  \bigl\{\,\mathfrak{U}\,\bigr\}
\]
by sending each configuration to its asynchronous limit.
\end{definition}

\begin{theorem}[Collapse as a Distributed Projection]
\[
\mathbf{Collapse}
\;\text{is a continuous, monotone, idempotent, globally defined projection.}
\]
\end{theorem}

\begin{proof}
Continuity follows from product-topological stability of the flow.
Monotonicity follows from strict Lyapunov descent.
Idempotence follows from Thm.~\ref{thm:async-idempotence}.
Global definition follows because every execution terminates
(Ch.~77).
\end{proof}

\section{Distributed Collapse Commutes with Networks}

\begin{theorem}[Network-Invariance]
\label{thm:network-invariance}
Let $\mathcal{N}$ and $\mathcal{N}'$ be arbitrary networks
(possibly with different topologies, communication delays, latencies,
or patch decompositions).  
Then for any global state $C$:
\[
\mathbf{Collapse}_{\mathcal{N}}(C)
\;=\;
\mathbf{Collapse}_{\mathcal{N}'}(C).
\]
\end{theorem}

\begin{proof}
All networks implement the same asynchronous operator~$\mathcal{A}$:
every valid scheduler corresponds to an interleaving of the same set of
primitive steps.  
Thus the limit (Thm.~\ref{thm:async-idempotence}) is independent of network.
\end{proof}

\noindent
This is the strongest universality result in Part~V:  
the semantics of the universe do not depend on the universe’s wiring.
