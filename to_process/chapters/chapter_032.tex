\chapter{Embedding Flows and Gradient Dynamics}
\label{ch:embedding-flows}

With the semantic embedding
$\iota : \mathcal{G} \hookrightarrow \mathcal{M}$
constructed in Chapter~\ref{ch:semantic-embedding}, we now study its
\emph{dynamics}.  
The continuous RSVP fields $(\Phi,\mathbf{v},S)$ evolve under their own
Lagrangian flow (Chs.~2–4), and the Polyxan hypergraph $\mathcal{G}$
evolves by curvature-decreasing rewriting (Chs.~22–26).  
When these systems are coupled, their joint evolution is governed by the
$L^2$–gradient flow of the \emph{total coupled energy}

\begin{equation}
\label{eq:E-total}
\mathcal{E}_{\mathrm{total}}[\Phi,\mathbf{v},S;\mathcal{G},\iota]
  = \mathcal{E}_{\mathrm{RSVP}}[\Phi,\mathbf{v},S]
  + \mathcal{E}_{\mathrm{Polyx}}[\mathcal{G}]
  + \mathcal{E}[\iota],
\end{equation}

where $\mathcal{E}[\iota]$ is the embedding energy functional introduced
in Chapter~31.

This chapter introduces:
\begin{itemize}
\item the \emph{embedding flow} $\partial_t\iota$,  
\item the \emph{modified RSVP field flow} under embedding back-reaction,  
\item the \emph{joint RSVP–Polyxan gradient flow},  
\item the \emph{monotone central-charge functional} $c_{\mathrm{total}}$,  
\item global existence and convergence of the joint flow,  
\item and the \emph{Coupled Irreversibility Theorem}, which states that
every trajectory approaches a unique semantic galaxy.
\end{itemize}

\section{The Total Coupled Energy Landscape}

Let
$(\Phi(t),\mathbf{v}(t),S(t))$ evolve on $\mathcal{M}$,
$\mathcal{G}(t)$ evolve by EHR rewriting,
and $\iota(t)$ evolve in the stratified space of embeddings.

We work on the Hilbert manifold

\begin{equation}
\mathfrak{X}
 = 
  W^{1,2}(\mathcal{M})
  \times W^{1,2}(T\mathcal{M})
  \times W^{1,2}(\mathcal{M})
  \times \mathrm{TypHyper}^{\mathrm{fin}}
  \times W^{1,2}(\mathrm{Inc}(\mathcal{G}),\mathcal{M}),
\end{equation}

equipped with the $L^2$ metric.  
$\mathcal{E}_{\mathrm{total}}$ is coercive, lower semicontinuous, and
real-analytic on each finite-dimensional slice.

Thus the $L^2$ gradient flow

\begin{equation}
\partial_t(\Phi,\mathbf{v},S;\mathcal{G},\iota)
  = -\nabla \mathcal{E}_{\mathrm{total}}
\end{equation}

is well defined.

\section{Embedding Flow as Gradient Descent}

We recall the embedding energy from Chapter~31:

\begin{align}
\mathcal{E}[\iota]
  &=
  \alpha\!\int_{\mathrm{Inc}(\mathcal{G})}
     \lVert \nabla\iota\rVert^2 d\mu \\
  &\qquad + \beta\!\int_{\mathrm{Inc}(\mathcal{G})}
     (\mathcal{R}\circ\iota - \mathcal{H}_0[\mathcal{G}])^2 d\mu \\
  &\qquad + \gamma\!\int_{\mathrm{Inc}(\mathcal{G})}
     (S\circ\iota - \mathcal{H}_{\mathrm{tag}}[\mathcal{G}])^2 d\mu .
\end{align}

Taking its $L^2$-gradient yields the \emph{embedding flow}:

\begin{equation}
\label{eq:embedding-flow-main}
\partial_t\iota
  = \Delta_{\mathrm{Inc}}\iota
   - \beta\,(\mathcal{R}\circ\iota - \mathcal{H}_0[\mathcal{G}])\,\nabla\mathcal{R}
   - \gamma\,(S\circ\iota - \mathcal{H}_{\mathrm{tag}}[\mathcal{G}])\,\nabla S.
\end{equation}

This is a geometric flow on the stratified domain $\mathrm{Inc}(\mathcal{G})$:
\begin{itemize}
\item $\Delta_{\mathrm{Inc}}$ smooths the embedding,
\item the curvature term pulls $\iota$ into regions of matching semantic curvature,
\item the entropy term aligns RSVP entropy with tag entropy.
\end{itemize}

\section{Embedding Back-Reaction on RSVP Fields}

The RSVP field equations acquire a distributional stress–energy term
supported on the embedded Polyxan hypergraph.  
We write:

\begin{equation}
T^{\mathrm{tot}}_{ab}
 = T^{\mathrm{RSVP}}_{ab}
   + T^{\mathrm{embed}}_{ab}[\iota].
\end{equation}

The embedding-induced stress is:

\begin{equation}
\label{eq:embedding-stress-expanded}
T_{ab}^{\mathrm{embed}}
 = F(\iota)\,h_{ab} - 2\alpha\,(\nabla_a\iota)^T(\nabla_b\iota),
\end{equation}

where $h_{ab}$ is the first fundamental form of the embedding.

Thus the RSVP field flow (Ch.~3) becomes:

\begin{align}
\partial_t\Phi
  &= \Delta\Phi - \beta\nabla\cdot\mathbf{v}
     - \frac{\delta\mathcal{E}[\iota]}{\delta\Phi}
     + T_{00}^{\mathrm{embed}}, \\[4pt]
\partial_t\mathbf{v}
  &= \Delta\mathbf{v} - \nabla\Phi - \gamma\nabla S
     + \mathrm{div}(T^{\mathrm{embed}}), \\[4pt]
\partial_t S
  &= \alpha\Delta S
     + \gamma\nabla\cdot\mathbf{v}
     + \mathcal{H}_{\mathrm{tag}}[\mathcal{G}]\,\delta_{\iota(\mathcal{G})}.
\end{align}

This \emph{back-reaction} couples the fields to the evolving discrete
structure.

\section{Discrete Evolution of the Hypergraph}

The Polyxan hypergraph evolves by curvature-decreasing EHR rewriting:

\begin{equation}
\mathcal{G}(t+\epsilon)
  = \mathrm{EHR}\big(\mathcal{G}(t)\big),
\end{equation}

whenever the embedding flow produces a topology mismatch between
$\iota(\mathcal{G})$ and the curvature strata of $\mathcal{M}$.

This is the discrete counterpart of RSVP singularity resolution.

\section{Unified Flow System}

The full evolution is a coupled PDE–ODE–rewriting system:

\begin{align}
\partial_t(\Phi,\mathbf{v},S)
  &= F_{\mathrm{RSVP}}(\Phi,\mathbf{v},S)
   + F_{\mathrm{back}}[\iota,\mathcal{G}], \\
\partial_t\iota
  &= -\frac{\delta\mathcal{E}[\iota]}{\delta\iota}, \\
\mathcal{G}(t+\epsilon)
  &= \mathrm{EHR}(\mathcal{G}(t)).
\end{align}

This system is well-posed due to:
\begin{itemize}
\item parabolic smoothing on $(\Phi,\mathbf{v},S)$,
\item elliptic regularity on $\iota$,
\item finite termination of $\mathcal{G}$ via the Polyxan c-theorem.
\end{itemize}

\section{Monotonicity of the Total Central Charge}

Recall the definitions:
\begin{equation}
c_{\mathrm{RSVP}} = \int_{\mathcal{M}} R^2 + \lVert\nabla(\Phi,\mathbf{v},S)\rVert^2 dV_g,
\end{equation}
\begin{equation}
c_{\mathrm{Polyx}}
 = \log\det\Delta_{\mathcal{G}} + \sum_e\kappa_{\mathcal{G}}(e)^2.
\end{equation}

Define:
\begin{equation}
c_{\mathrm{total}}(t)
  = c_{\mathrm{RSVP}}(t) + c_{\mathrm{Polyx}}[\mathcal{G}(t)].
\end{equation}

\begin{theorem}[Total Central Charge Monotonicity]
\label{thm:c-total-monotone}
Along the coupled embedding flow,
\begin{equation}
\frac{d}{dt}c_{\mathrm{total}}(t)
  = -\int_{\mathcal{M}}\lVert\nabla(\Phi,\mathbf{v},S)\rVert^2 dV_g
    - \sum_{e\in\mathcal{G}(t)}
       \bigl|\nabla\kappa_{\mathcal{G}(t)}(e)\bigr|^2
    \le 0.
\end{equation}
Equality holds iff the system is at a semantic galaxy.
\end{theorem}

\begin{proof}
Differentiating the RSVP term yields a non-positive integral by the
entropy-production inequality (Ch.~3).  
Differentiating the Polyxan term yields a non-positive curve due to the
Polyxan c-theorem (Ch.~23).
\end{proof}

Thus the system evolves irreversibly toward a unique fixed point.

\section{Finite-Time Galaxy Formation}

The continuous RSVP flow collapses hyperbolic regions to elliptic
regions in finite time (Ch.~13), while the embedding flow collapses
high-curvature embeddings into harmonic ones.

Simultaneously, the Polyxan hypergraph collapses via EHR rewriting into
a canonical media quine.

\begin{theorem}[Finite-Time Galaxy Formation]
\label{thm:galaxy-formation}
For any compactly supported initial condition, the coupled
RSVP–Polyxan system produces a bounded semantic galaxy in finite time.
\end{theorem}

\begin{proof}
Hyperbolic RSVP modes collapse in finite time (Ch.~13).  
Embedding curvature collapses under~\eqref{eq:embedding-flow-main}.  
Discrete curvature collapses by the Polyxan c-theorem.  
All domains become elliptic and bounded, enforcing termination.
\end{proof}

\section{The Coupled Irreversibility Theorem}

We now state the central result of Part~III.

\begin{theorem}[Coupled Irreversibility Theorem]
Every trajectory of the coupled RSVP–Polyxan embedding flow converges to
a unique semantic galaxy $\mathfrak{G}$ characterised by:
\begin{itemize}
\item vanishing total curvature $\mathcal{R}=0$ on $\iota(\mathcal{G})$,  
\item vanishing positive-degree cohomology $H^k(\mathcal{G})=0$ for $k>0$,  
\item harmonic RSVP fields $(\Phi^\infty,0,S^\infty)$,  
\item and a flat, harmonic embedding of the Polyxan skeleton.
\end{itemize}
\end{theorem}

\begin{proof}
Monotonicity of $c_{\mathrm{total}}$ gives existence of a limit.  
Uniqueness of the media quine (Ch.~26) and uniqueness of harmonic embeddings
(Ch.~31) give convergence to a unique galaxy.  
\end{proof}

This completes the theory of embedding flows.  
Semantic galaxies are now fully dynamical objects: born in finite time,
stable, and unique.
