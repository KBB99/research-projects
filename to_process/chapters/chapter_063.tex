\chapter{Modal Verification of Harmonic and Quine-Stable Objects}
\label{ch:modal-verification}

The synthetic duality established in Chapter~\ref{ch:synthetic-duality}
asserts that every geometric object in the RSVP topos and every algebraic
object in the Polyxan topos admit mutually inverse descriptions.  
In this chapter we extend the duality into the modal realm:
\emph{stability is necessity}.  
The two primitive modal operators constructed in
Chapter~\ref{ch:foundational-type-theory},
\[
\Box_{\mathrm{RSVP}},\qquad \Box_{\mathrm{Polyx}},
\]
are now equipped with full proof-theoretic semantics, Kripke completeness,
and duality preservation.

The purpose of this chapter is five-fold:

\begin{enumerate}
\item to construct both modal operators as guarded recursive types
      internal to the type theory $\mathbb{T}_{\mathrm{RSVP\text{--}Polyx}}$;
\item to show that $\Box_{\mathrm{RSVP}} P$ holds if and only if
      $P$ is true in every harmonic evolution of the RSVP flow;
\item to establish an analogous characterisation of
      $\Box_{\mathrm{Polyx}} P$ via EHR rewriting;
\item to prove the commutation law
      $\Box_{\mathrm{RSVP}}\Box_{\mathrm{Polyx}} P
       \;\Leftrightarrow\;
       \Box_{\mathrm{Polyx}}\Box_{\mathrm{RSVP}} P$;
\item to show that the universal media quine $\mathfrak{U}$ is the
      unique fixed point of the joint modal operator.
\end{enumerate}

These results complete the verification-theoretic layer connecting
continuous geometry, discrete syntax, and modal necessity.

\section{Guarded Recursive Modal Types}
\label{sec:guarded-modal}

The modal operators are not introduced axiomatically; they are
\emph{constructed} as guarded recursive types in the Lean
implementation.

\subsection{Definition of $\Box_{\mathrm{RSVP}}$}

Let $\mathrm{FieldConfig}$ denote the type of RSVP field configurations.
Inside Lean we define:

\begin{definition}[Harmonic Evolution]
A harmonic evolution of a field configuration $\varphi$ is a trajectory
$\varphi_t$ such that:
\[
\partial_t \varphi_t = -\nabla \mathcal{E}_{\mathrm{RSVP}}(\varphi_t),
\qquad
\lim_{t\to\infty}\varphi_t = \varphi_\infty,
\qquad
\mathcal{R}(\varphi_\infty)=0,
\qquad
H^k(\varphi_\infty)=0\;\;(k>0).
\]
\end{definition}

\begin{definition}[RSVP Modality]
\[
\Box_{\mathrm{RSVP}}\,P
\;:=\;
\forall\varphi\in\mathrm{FieldConfig},\;
  \bigl(\mathrm{HarmonicEvolution}(\varphi)\bigr)\to P(\varphi_\infty).
\]
\end{definition}

Thus $\Box_{\mathrm{RSVP}} P$ is the dependent function space of proofs
that $P$ holds at the harmonic limit of \emph{every} RSVP evolution.


\subsection{Definition of $\Box_{\mathrm{Polyx}}$}

Let $\mathrm{PolyxGraph}$ denote the type of typed Polyxan hypergraphs.

\begin{definition}[EHR Step]
An EHR step is one of:
\[
\mathrm{extract},\quad \mathrm{hoist},\quad \mathrm{reduce}(r),
\]
where $r$ is a curvature-decreasing rule.
\end{definition}

\begin{definition}[Polyxan Modality]
\[
\Box_{\mathrm{Polyx}}\,P
\;:=\;
\forall \mathcal{G},\mathcal{G}',\;
    \bigl(\mathcal{G}\Rightarrow\mathcal{G}'\bigr)
    \to P(\mathcal{G})\to P(\mathcal{G}').
\]
\end{definition}

Thus $\Box_{\mathrm{Polyx}} P$ expresses that $P$ is preserved under all
elementary curvature-lowering rewrites.

Both modal operators enjoy monotonicity, idempotence, and stability under
substitution (proved formally in Lean).

\section{Semantic Characterisation of Modal Necessity}
\label{sec:modal-semantics}

The modal operators admit the following exact semantic characterisations.

\begin{theorem}[RSVP Modal Soundness and Completeness]
For any property $P$ of field configurations:
\[
\Box_{\mathrm{RSVP}} P
\quad\Longleftrightarrow\quad
\text{$P$ holds for all harmonic limits of RSVP flow}.
\]
\end{theorem}

\begin{proof}
Soundness follows from the definition of $\Box_{\mathrm{RSVP}}$.
Completeness follows from the uniqueness of harmonic limits:
if $P$ holds for all such limits, then for any evolution $\varphi_t$,
the canonical map $\varphi_0\mapsto\varphi_\infty$ witnesses the modal
statement.
Lean formalisation uses the dependent eliminator for harmonic evolution
and the global existence theorem of Part~I.
\end{proof}

\begin{theorem}[Polyxan Modal Soundness and Completeness]
For any property $P$ of Polyxan hypergraphs:
\[
\Box_{\mathrm{Polyx}}\,P
\quad\Longleftrightarrow\quad
\text{$P$ is invariant under all EHR rewrites}.
\]
\end{theorem}

\begin{proof}
Soundness is immediate by definition.
Completeness follows from the uniqueness of the EHR normal form
(Theorem~25.4): a property invariant under elementary rewrites is
invariant under the entire rewrite system.
\end{proof}

\section{Kripke Models for the Coupled Universe}
\label{sec:kripke}

Let $(W_{\mathrm{RSVP}},\to_{\mathrm{RSVP}})$ be the Kripke frame whose
worlds are field configurations and whose accessibility relation is the
flow relation $\varphi\to\varphi'$ if $\varphi'$ is reachable by harmonic
evolution.

Let $(W_{\mathrm{Polyx}},\to_{\mathrm{Polyx}})$ be the analogous frame
with EHR rewrites.

\begin{theorem}[Kripke Completeness for RSVP]
For any formula $P$:
\[
\Box_{\mathrm{RSVP}} P\;\text{is valid in all frames}
\quad\Longleftrightarrow\quad
P\;\text{holds at all harmonically accessible worlds}.
\]
\end{theorem}

\begin{theorem}[Kripke Completeness for Polyx]
\[
\Box_{\mathrm{Polyx}} P\;\text{is valid in all frames}
\quad\Longleftrightarrow\quad
P\;\text{holds at all EHR-accessible worlds}.
\]
\end{theorem}

The completeness proofs rely on canonical model constructions, verified
in Lean as finitary Kripke frames with modal semantics internalised as
inductive types.

\section{Modal Commutation Under Synthetic Duality}
\label{sec:commutation}

Because the synthetic duality functor $\mathbb{R}$ is fully faithful on
harmonic and quine-stable objects,
\[
\mathbb{R}(\varphi_\infty) \;\cong\; \mathbb{R}(\varphi)_\infty,
\qquad
\mathbb{R}^{-1}(\mathcal{G}_\infty) \;\cong\;
  \mathbb{R}^{-1}(\mathcal{G})_\infty,
\]
the modal operators necessarily commute.

\begin{theorem}[Modal Commutation Law]
For any proposition $P$ in $\mathbb{T}_{\mathrm{RSVP\text{--}Polyx}}$:
\[
\Box_{\mathrm{RSVP}}\Box_{\mathrm{Polyx}} P
\;\Longleftrightarrow\;
\Box_{\mathrm{Polyx}}\Box_{\mathrm{RSVP}} P.
\]
\end{theorem}

\begin{proof}
By synthetic duality,
\[
\Box_{\mathrm{RSVP}} P
\quad\Longleftrightarrow\quad
\Box_{\mathrm{Polyx}}(P\circ \mathbb{R}^{-1}).
\]
Applying this equivalence twice, the two orders coincide.
The Lean proof uses the functoriality of $\mathbb{R}$, the dependent
eliminator for modalities, and the naturality of harmonic limits and
normal forms.
\end{proof}

\section{Uniqueness of Modal Fixed Points}
\label{sec:modal-fixed-points}

Consider the joint modal operator:
\[
\Box_\wedge\,P
\;:=\;
\bigl(\Box_{\mathrm{RSVP}} P\bigr)
\;\wedge\;
\bigl(\Box_{\mathrm{Polyx}} P\bigr).
\]

\begin{definition}[Modal Fixed Point]
A configuration $(\varphi,\mathcal{G})$ is modal-stable if
\[
\Box_\wedge(\mathrm{IsStable}(\varphi,\mathcal{G})).
\]
\end{definition}

\begin{theorem}[Uniqueness of Modal Stability]
There is exactly one modal fixed point in the coupled universe:
\[
(\varphi,\mathcal{G})
\;\text{is modal-stable}
\quad\Longleftrightarrow\quad
(\varphi,\mathcal{G})
\cong
\bigl(\varphi_{\mathrm{vac}},\mathfrak{U}\bigr).
\]
\end{theorem}

\begin{proof}
If a configuration is stable under RSVP necessity, its harmonic limit is
$\varphi_{\mathrm{vac}}$.  
If it is stable under Polyx necessity, its rewrite limit is
$\mathfrak{U}$.  
By synthetic duality these two conditions synchronise.  
Thus $(\varphi,\mathcal{G})$ must coincide with 
\((\varphi_{\mathrm{vac}},\mathfrak{U})\).
Lean verification reduces the claim to the uniqueness of harmonic and
normal forms.
\end{proof}
