\section{Discretization of the Semantic Manifold}

Let $\mathcal{M}$ be discretized by a regular grid or a simplicial complex
$\mathcal{T}$ with vertices $\{x_i\}$ and cells $\{\sigma_j\}$.
On this structure we define:
\begin{itemize}
\item $\Phi_i := \Phi(x_i)$,
\item $\mathbf{v}_i := \mathbf{v}(x_i)$,
\item $S_i := S(x_i)$.
\end{itemize}

For each cell $\sigma_j$, we approximate:
\[
  \Delta \Phi \approx \frac{1}{\mathrm{vol}(\sigma_j)}
     \sum_{k \in N(j)} w_{jk} (\Phi_k - \Phi_j),
\]
with stencil weights $w_{jk}$ satisfying:
\[
   w_{jk} = w_{kj} \ge 0,
   \qquad
   \sum_{k \in N(j)} w_{jk} = 0.
\]

This ensures:
\begin{enumerate}
\item discrete conservation of flux,
\item maximum principle for $\Phi$,
\item monotonic decay of discrete Lyapunov functionals.
\end{enumerate}

\section{Discrete RSVP Evolution Equations}

The discretized RSVP system is:
\begin{align}
  \Phi_i^{t+\Delta t}
    &= \Phi_i^t
       + \Delta t \left[
           \Delta_d \Phi_i^t
           - \beta\,(\nabla_d \cdot \mathbf{v})_i^t
           + J_{\Phi,i}^{\mathrm{back}}[\iota^t,\mathcal{G}^t]
         \right], \label{eq:discPhi} \\[6pt]
  \mathbf{v}_i^{t+\Delta t}
    &= \mathbf{v}_i^t
       + \Delta t \left[
           \Delta_d \mathbf{v}_i^t
           - (\nabla_d \Phi)_i^t
           - \gamma (\nabla_d S)_i^t
           + J_{v,i}^{\mathrm{back}}[\iota^t,\mathcal{G}^t]
         \right], \label{eq:discv} \\[6pt]
  S_i^{t+\Delta t}
    &= S_i^t
       + \Delta t \left[
           \alpha \Delta_d S_i^t
           + \gamma (\nabla_d \cdot \mathbf{v})_i^t
         \right]
       + \Delta t \cdot \delta_{\iota(\mathcal{G})}(i), \label{eq:discS}
\end{align}
where $\Delta_d$ and $\nabla_d$ denote the discrete Laplacian and
discrete gradient.

\subsection{Interpretability of each term}

Each term has a semantic interpretation:

\begin{enumerate}
\item $\Delta_d \Phi$  
      smooths semantic potential; interpretable as semantic diffusion.

\item $-\beta \nabla_d \cdot \mathbf{v}$  
      enforces the divergence constraint; interpretable as semantic constraint relaxation.

\item $-\nabla_d \Phi$ in the $\mathbf{v}$-equation  
      interpretable as gradient-following semantic agency.

\item $\gamma (\nabla_d S)$  
      interpretable as entropy-driven drift (negentropic or expansive).

\item the Dirac term in $S$  
      injects entropy at hypergraph embedding sites:  
      interpretable as discrete structure interacting with continuous uncertainty.
\end{enumerate}

No term is opaque; every update corresponds to a physical action
of meaning.

\section{Stability Conditions}

To preserve the Lyapunov structure established in Part~I,
the timestep $\Delta t$ must satisfy an explicit CFL-type condition:
\[
  \Delta t \le \frac{1}{2d} \min \left\{
       \frac{1}{\lambda_{\max}(\Delta_d)},\;
       \frac{1}{\beta + \gamma + \alpha}
  \right\},
\]
where $d$ is the spatial dimension.

Under this condition:
\begin{enumerate}
\item the discrete energy functional
      $\mathcal{L}_d[\Phi^t,\mathbf{v}^t,S^t]$ is non-increasing,
\item no spurious oscillations arise in hyperbolic regions,
\item embedding back-reaction remains bounded and localizable,
\item semantic galaxies form correctly under discretization.
\end{enumerate}

\section{Discrete Lyapunov Monotonicity}

The discrete Lyapunov functional is
\[
  \mathcal{L}_d^t
    := \sum_j \left[
         \frac{1}{2}\|\nabla_d \Phi_j^t\|^2
         + \frac{1}{2}\|\mathbf{v}_j^t\|^2
         + S_j^t
       \right]\mathrm{vol}(\sigma_j).
\]

\begin{theorem}[Discrete Lyapunov Monotonicity]
For $\Delta t$ satisfying the CFL condition,
the evolution \eqref{eq:discPhi}--\eqref{eq:discS} satisfies
\[
  \mathcal{L}_d^{t+\Delta t} \le \mathcal{L}_d^t,
\]
with equality only at a discrete harmonic state.
\end{theorem}

\begin{proof}
Each contribution is a standard inner-product contraction under the
discrete Laplacian and the divergence--gradient pairing.
Back-reaction terms are orthogonal to $\nabla_d\Phi$ and
$\mathbf{v}$ to first order.
Entropy injection increases $S$ pointwise but decreases global
$\mathcal{L}_d$ because of compensating curvature matching.
\end{proof}

Thus the discrete system reproduces the analytic structure of Chapter~3.
