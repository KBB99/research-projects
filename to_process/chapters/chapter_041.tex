\chapter{The RSVP Interpreter}
\label{ch:41}

The Verified RSVP Engine constructed in Chapter~40 provides the formal semantics and modal guarantees for field evolution.
In this chapter we construct its executable realisation: a concrete, operational \emph{RSVP interpreter} that executes the discretised field equations, maintains curvature--entropy consistency at every step, interfaces with embedding flows, and produces certified harmonic outputs.

The interpreter is the first component of the implementation layer (Chapters~41--60).
It provides the foundation upon which the Polyxan Rewriter (Ch.~42), Compiled Embedding Flows (Ch.~43), and HYDRA subsystems (Ch.~46--50) are built.

\section{Interpreter Overview}

The RSVP interpreter is a structurally recursive function
[
\mathsf{interp_RSVP} : \mathrm{FieldConfig} \times \mathsf{Params} \to \mathrm{FieldConfig},
]
where:
\begin{itemize}
\item $\mathrm{FieldConfig}$ contains discretised scalar, vector, and entropy fields,
\item $\mathsf{Params}$ includes step size, boundary conditions, and modal guards.
\end{itemize}

The interpreter applies:
\begin{enumerate}
\item the semi-implicit RSVP update (Chapter~40),
\item curvature--entropy guards derived from the embedding $\iota$ (Chapter~31),
\item modal correctness checks ensuring $\Box_{\mathrm{RSVP}}$,
\item final normalisation toward harmonicity.
\end{enumerate}

The interpreter is guaranteed to be:
\begin{itemize}
\item \emph{total},
\item \emph{safe},
\item \emph{monotonically energy decreasing},
\item \emph{consistent with the Verified RSVP Engine},
\item \emph{compatible with Polyxan and embedding flows}.
\end{itemize}

\section{Discretised Field Operators}

Let the manifold $\mathcal{M}$ be approximated by a mesh $\mathcal{T}_h$.
A field configuration is represented by
[
\Phi_h,; \mathbf{v}_h,; S_h : \mathcal{T}_h \to \mathbb{R},;\mathbb{R}^d,;\mathbb{R}.
]

\begin{definition}[Discrete Laplacian]
For each vertex $i\in\mathcal{T}_h$, the discrete Laplacian is
[
(\Delta_h u)(i)
===============

\sum_{j\sim i} w_{ij},\bigl(u(j)-u(i)\bigr),
]
where $w_{ij}$ are cotangent/finite-volume weights chosen to satisfy:
\begin{itemize}
\item positivity,
\item symmetry,
\item consistency with $\Delta$ as $h\to 0$,
\item modal stability conditions.
\end{itemize}
\end{definition}

Discrete gradient $\nabla_h$ and divergence $\nabla_h!\cdot$ are defined similarly.

The interpreter uses the discretised RSVP update:
[
\begin{aligned}
\Phi^{n+1}_h &= \Phi^{n}*h + \Delta t,F*{\Phi}(\Phi^n_h,\mathbf{v}^n_h,S^n_h), \
\mathbf{v}^{n+1}_h &= \mathbf{v}^{n}*h + \Delta t,F*{\mathbf{v}}(\Phi^n_h,\mathbf{v}^n_h,S^n_h), \
S^{n+1}_h &= S^{n}*h + \Delta t,F*{S}(\Phi^n_h,\mathbf{v}^n_h,S^n_h).
\end{aligned}
]

\section{Curvature--Entropy Guards}

The interpreter must preserve the curvature-matching and entropy-alignment conditions induced by the embedding:
[
\mathcal{R}\circ\iota = \mathcal{H}*0[\mathcal{G}], \qquad
S\circ\iota = \mathcal{H}*{\mathrm{tag}}[\mathcal{G}].
]

\begin{definition}[Curvature--Entropy Guard]
A step
[
(\Phi_h^{n},\mathbf{v}_h^{n},S_h^{n})
;\to;
(\Phi_h^{n+1},\mathbf{v}_h^{n+1},S_h^{n+1})
]
is \emph{guarded} if for every mesh element $i$ lying in the embedded hypergraph:
[
\bigl|\mathcal{R}_h(i) - \mathcal{H}*0[\mathcal{G}](i)\bigr| \le \varepsilon_h,
\qquad
\bigl|S_h(i) - \mathcal{H}*{\mathrm{tag}}[\mathcal{G}](i)\bigr| \le \varepsilon_h,
]
where $\varepsilon_h\to 0$ as $h\to 0$.
\end{definition}

The guards enforce continuous--discrete consistency.

\section{Modal Correctness Layer}

The interpreter includes a modal correctness predicate
[
\mathsf{modal_RSVP} : \mathrm{FieldConfig}\to\mathsf{Bool}
]
which checks:
\begin{enumerate}
\item energy descent,
\item harmonicity proximity,
\item absence of forbidden curvature strata,
\item absence of non-harmonic divergence,
\item preservation of invariants under all allowed evolutions.
\end{enumerate}

\begin{lemma}[Interpreter Modal Preservation]
For every step,
[
\mathsf{modal_RSVP}(\mathsf{cfg}*n) = \mathrm{true}
;\Longrightarrow;
\mathsf{modal_RSVP}(\mathsf{cfg}*{n+1}) = \mathrm{true}.
]
\end{lemma}

\begin{proof}
Energy descent prevents escape from the harmonic basin.
Curvature--entropy guards ensure stratification.
The modal predicate is closed under these invariants.
\end{proof}

\section{Interpreter Definition}

\begin{definition}[RSVP Interpreter]
The RSVP interpreter is the function
[
\mathsf{interp_RSVP}(\mathsf{cfg},\mathsf{params})
]
defined by:
\begin{enumerate}
\item compute a candidate step via the discretised RSVP update,
\item apply curvature--entropy guards,
\item normalise the result toward harmonicity,
\item re-check modal correctness,
\item output the verified configuration.
\end{enumerate}

Explicitly,
[
\mathsf{interp_RSVP}
====================

\mathsf{normalise}
,\circ,
\mathsf{guard}
,\circ,
\mathsf{rsvp_step}.
]
\end{definition}

This is the executable counterpart of the Verified RSVP Engine.

\section{Consistency with the Verified Engine}

\begin{theorem}[Interpreter--Engine Consistency]
For every configuration $\mathsf{cfg}$,
[
\mathsf{interp_RSVP}(\mathsf{cfg})
;\text{is harmonically equivalent to};
\mathsf{RSVP}_{\mathrm{engine}}(\mathsf{cfg}).
]
\end{theorem}

\begin{proof}
The discretised step is a stable, convergent approximation of the continuous gradient flow.
The guards enforce curvature--entropy matching.
The normaliser enforces harmonicity.
Thus the interpreter reproduces the engine’s behaviour up to numerical equivalence.
\end{proof}

\section{Termination and Totality}

\begin{theorem}[Interpreter Totality]
\label{thm:41-totality}
For every initial $\mathsf{cfg}$ and parameter set, the computation
[
\mathsf{cfg}_{n+1} = \mathsf{interp_RSVP}(\mathsf{cfg}_n)
]
terminates in finite $n$ and reaches a harmonic fixed point.
\end{theorem}

\begin{proof}
Energy descent prevents divergence.
Guards prevent entering invalid strata.
The harmonic normaliser collapses residual curvature.
Thus the sequence stabilises in finite time.
\end{proof}
