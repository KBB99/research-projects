\chapter{The Cognitive Lagrangian and Rewrite Dynamics}
\label{ch:cognitive-lagrangian}

\section{Overview}

Chapters~\ref{ch:semantic-manifold}--\ref{ch:lsystem} developed the geometric,
thermodynamic, and rewrite-theoretic structure of the RSVP--Polyxan semantic
universe. In the present chapter we introduce a unifying variational framework:
the \emph{Cognitive Lagrangian}. This Lagrangian governs the evolution of
semantic configurations under rewrite operations and ensures consistency with
RSVP thermodynamics, sheaf morphisms, entropy bounds, and the BV master
equation.

The Cognitive Lagrangian provides:
\begin{enumerate}
  \item an energy functional for rewrite expansions, contractions, and
        substitutions;
  \item a metric on semantic configurations, derived from RSVP gradients;
  \item an action principle unifying smoothing, generativity, and intentionality;
  \item an induced system of Euler--Lagrange equations describing rewrite flow;
  \item and a BV extension ensuring homotopy-consistent and gauge-invariant
        semantics.
\end{enumerate}

This layer makes the rewrite dynamics compatible with physical-style
variational principles and enables connections with AKSZ-type sigma models,
entropy descent, and verification formalisms (cf.\
Chapters~\ref{ch:coupling} and~\ref{ch:verification}).

\section{Configuration Space and Kinematics}

Recall the configuration category:
\[
\mathsf{Conf}(\mathcal{M})
\]
whose objects are semantic configurations
\[
X \in \Gamma(U, \mathscr{P}^{\otimes n}),
\]
and whose morphisms are restriction maps and natural transformations.

\subsection{Tangent structure on configuration space}

We equip $\mathsf{Conf}(\mathcal{M})$ with a tangent-like structure:
\[
T_X\mathsf{Conf}(\mathcal{M}) := \Gamma(U, \mathscr{P}^{\otimes n})
\]
consisting of infinitesimal variations $\delta X$ supported on the same domain.

These variations capture:
\begin{enumerate}
  \item small geometric displacements,
  \item small changes in Polyxan atom types,
  \item first-order rewrite expansions.
\end{enumerate}

\subsection{Semantic kinetic energy}

The RSVP-induced metric $g$ from Chapter~\ref{ch:rsvp-fields} endows tangent
variations with a scalar product:
\[
\|\delta X\|_g^2
:=
\int_U
g^{ij}\langle \partial_i \delta X, \partial_j \delta X\rangle \,\mathrm{dvol}_g.
\]

This defines the kinetic term for the Cognitive Lagrangian.

\section{The Cognitive Potential and Rewrite Energy}

Rewrite rules $\rho$ (Chapter~\ref{ch:lsystem}) introduce configurational
expansion and branching. Each rewrite step has an intrinsic \emph{rewrite
energy}.

\subsection{Local rewrite energy density}

Given a rewrite rule $\rho: \mathscr{P}\to\mathscr{P}^{\otimes k}$ and a local
atom $p\in\mathscr{P}_x$, define:

\begin{definition}[Rewrite Energy Density]
\[
\mathcal{E}_\rho(p)
:=
\alpha\,\|\rho(p)-p\|_g^2
+
\beta\,(S(\rho(p))-S(p))
+
\gamma\,(q_v(\rho(p)) - q_v(p))\cdot \mathbf{v}.
\]
\end{definition}

Interpretation:
\begin{itemize}
  \item $\alpha$ controls geometric distortion,
  \item $\beta$ encodes entropy flow,
  \item $\gamma$ weights directional intentionality.
\end{itemize}

\subsection{Global rewrite potential}

For a configuration $X$, define:
\[
V_{\rho}(X)
=
\int_U \mathcal{E}_\rho\big(X(x)\big)\,\mathrm{dvol}_g.
\]

The potential energy captures the “cost” of rewriting $X$ using $\rho$.

\section{The Cognitive Lagrangian}

\subsection{Action functional}

The \emph{Cognitive Lagrangian} for a path
$X: [0,T] \to \mathsf{Conf}(\mathcal{M})$ is:
\[
\mathcal{L}[X, \dot{X}]
=
\int_U
\left[
\frac{1}{2} \|\dot{X}\|_g^2
-
\sum_{\rho\in\mathcal{R}} \lambda_\rho\, \mathcal{E}_\rho(X)
-
W(X; \Phi,S,\mathbf{v})
\right]\mathrm{dvol}_g.
\]

The term $W$ encodes \emph{RSVP coupling}:
\[
W(X) :=
\mu_1\,\langle\nabla\Phi, \nabla X\rangle
+
\mu_2\, S\,\mathrm{div}(X)
+
\mu_3\,\mathbf{v}\cdot\nabla X.
\]

\subsection{Cognitive action}

Define:
\[
\mathcal{S}_{\mathrm{cog}}[X]
=
\int_0^T \mathcal{L}[X,\dot{X}]\, \mathrm{d}t.
\]

A cognitive trajectory is a critical point:
\[
\delta \mathcal{S}_{\mathrm{cog}} = 0.
\]

\section{Euler--Lagrange Equations for Rewrite Dynamics}

Compute:
\[
\frac{\delta\mathcal{S}_{\mathrm{cog}}}{\delta X}
=
0.
\]

\subsection{Kinetic contribution}
\[
-\nabla_\mu \left( g^{\mu\nu} \partial_\nu \dot{X} \right)
\]

\subsection{Rewrite potential gradients}
\[
+\sum_{\rho} \lambda_\rho \,\frac{\partial \mathcal{E}_\rho}{\partial X}
\]

\subsection{RSVP coupling contributions}
\[
-\mu_1\,\nabla_\mu\nabla^\mu \Phi
-\mu_2\,\nabla_\mu(S\,\delta^\mu X)
-\mu_3\,\mathcal{L}_{\mathbf{v}} X
\]

\subsection{Full equation}

\[
\nabla_\mu \nabla^\mu X
=
\sum_{\rho} \lambda_\rho \nabla_X \mathcal{E}_\rho
+
\mu_1 \nabla_\mu\nabla^\mu \Phi
+
\mu_2 \nabla_\mu(S\,\delta^\mu X)
+
\mu_3 \mathcal{L}_{\mathbf{v}}X.
\]

This is the \emph{cognitive wave--diffusion equation}, coupling rewrite
propagation to RSVP geometry and entropy smoothing.

\section{BV and AKSZ Structure}

Rewrite processes possess gauge-like redundancies generated by:
\begin{itemize}
  \item diffeomorphisms of $\mathcal{M}$,
  \item reparametrizations of configuration space,
  \item equivalence of rewrite paths via monoidal coherence.
\end{itemize}
These require a BV formulation.

\subsection{Ghosts and antifields}

Introduce:
\[
X^*,\qquad c^\mu,\qquad c_\mu^*,
\]
with standard ghost number assignments.

\subsection{AKSZ-type BV action}

Define:
\[
\mathcal{S}_{\mathrm{BV}}
=
\mathcal{S}_{\mathrm{cog}}
+
\int_U
\left[
X^*\,\mathcal{L}_c X
+
c_\mu^*(c\cdot\nabla c)^\mu
\right]\mathrm{dvol}_g.
\]

\begin{proposition}[Classical Master Equation]
The action $\mathcal{S}_{\mathrm{BV}}$ satisfies:
\[
\{\mathcal{S}_{\mathrm{BV}}, \mathcal{S}_{\mathrm{BV}}\}=0.
\]
\end{proposition}

\begin{proof}[Sketch]
Rewrite transformations, metric symmetries, and diffeomorphisms form a closed
algebra under the BV antibracket. Nilpotence follows from the Jacobi identity
and the naturality of $\rho$ under $\mathrm{Aut}(\mathcal{M})$.
\end{proof}

\section{Gradient-Flow Formulation}

Define:
\[
\frac{\delta \mathcal{S}_{\mathrm{cog}}}{\delta X}
=
\operatorname{grad}\mathcal{S}_{\mathrm{cog}}.
\]

Then cognitive morphogenesis can be written:
\[
\partial_t X = -\,\operatorname{grad}\mathcal{S}_{\mathrm{cog}}.
\]

\begin{definition}[Cognitive Gradient Flow]
\[
\dot{X}
=
\nabla_\mu \nabla^\mu X
-
\sum_{\rho}\lambda_\rho\,\nabla_X\mathcal{E}_\rho
-
\mu_1\nabla_\mu\nabla^\mu\Phi
-
\mu_2\nabla_\mu(S\,\delta^\mu X)
-
\mu_3\mathcal{L}_{\mathbf{v}}X.
\]
\end{definition}

This form reveals the deep analogy between rewrite morphogenesis and entropy
smoothing on the semantic manifold.

\section{Energy Bounds and Semantic Curvature}

Rewrite evolution must respect RSVP curvature--entropy constraints.

\begin{proposition}[Semantic Energy Bound]
Let $\mathcal{K}$ be the semantic curvature scalar of $g$. Then any admissible
rewrite trajectory must satisfy
\[
\|\dot{X}\|_g^2
\le
C_1 + C_2 \bigl(|\mathcal{K}| + \|\nabla\Phi\|^2 + S\bigr)
\]
for constants $C_1,C_2>0$.
\end{proposition}

\begin{proof}[Sketch]
Follows by combining convexity of the rewrite potential with boundedness of
the RSVP Lagrangian from below (Chapter~\ref{ch:rsvp-fields}) and ellipticity
of the Laplacian on each stratum of the semantic manifold.
\end{proof}
