\label{ch:synthetic-duality}

The unification achieved analytically in Parts~I--III is now re-expressed as a

\mathbb{R} : \mathbf{RSVP} \;\rightleftarrows\; \mathbf{Polyx} : \mathbb{R}^{-1},

\section{The Dual Topoi of RSVP and Polyxan Structure}

Let
\[
\mathbf{RSVP} := \mathbf{Sh}(\mathcal{M})
\]
be the topos of smooth RSVP field configurations:
sheaves of jet--bundle sections equipped with the RSVP flow.

Let
\[
\mathbf{Polyx} := \mathbf{HSh}(\mathrm{Strat})
\]
be the topos of Polyxan hypergraphs:
stratified higher sheaves with EHR rewrite dynamics.

Both are coherent, locally presentable, and admit internal homotopy types.
All constructions are internalised in
$\mathbb{T}_{\mathrm{RSVP\text{--}Polyx}}$ (Chapter~61).

\section{Definition of the Synthetic Duality Functor}

\subsection{Forward Functor \texorpdfstring{$\mathbb{R}$}{R} : Geometry → Syntax}

For an RSVP configuration $(\Phi,\mathbf{v},S)$,
the functor $\mathbb{R}$ acts by:

\begin{enumerate}
\item extracting the stratified critical set of $(\Phi,\mathbf{v},S)$,
\item forming its incidence hypergraph using RSVP curvature as hyperedge type,
\item assigning entropy tags via $S$,
\item equipping the result with EHR rewrite dynamics matching the RSVP energy functional.
\end{enumerate}

Formally,
\[
\mathbb{R}(\Phi,\mathbf{v},S)
  := \mathrm{Crit}_{\mathrm{strat}}(\Phi,\mathbf{v},S)
     \;\text{with curvature/entropy labels}.
\]

Morphisms are pushed forward via energy--decreasing graph maps.

\subsection{Inverse Functor \texorpdfstring{$\mathbb{R}^{-1}$}{R⁻¹} : Syntax → Geometry}

For a Polyxan hypergraph $\mathcal{G}$,
we define:
\[
\mathbb{R}^{-1}(\mathcal{G})
  := (\Phi_{\mathcal{G}},\mathbf{v}_{\mathcal{G}},S_{\mathcal{G}}),
\]
the unique RSVP field triple satisfying:

\begin{align}
\mathcal{R}\circ \iota_{\mathcal{G}} &= \mathcal{H}_0[\mathcal{G}], \\
S\circ \iota_{\mathcal{G}} &= \mathcal{H}_{\mathrm{tag}}[\mathcal{G}], \\
\Delta \Phi_{\mathcal{G}} &= \nabla\cdot\mathbf{v}_{\mathcal{G}} + \text{EHR source},
\end{align}

where $\iota_{\mathcal{G}}$ is the canonical semantic embedding (Ch.~31)
and the source term accounts for curvature--reducing reductions.

This is constructively definable in the unified type theory (Ch.~61)
as a terminating recursive computation.

\section{Full Faithfulness on Stable Objects}

\begin{theorem}[Full Faithfulness on Harmonic/Quine--Stable Objects]
\label{thm:full-faithful}
Restricted to harmonic RSVP fields and quine--stable Polyxan hypergraphs,
$\mathbb{R}$ is fully faithful:
\[
\mathrm{Hom}_{\mathbf{RSVP}}(\varphi_1,\varphi_2)
  \;\cong\;
\mathrm{Hom}_{\mathbf{Polyx}}(\mathbb{R}\varphi_1,\mathbb{R}\varphi_2).
\]
\end{theorem}

\begin{proof}[Sketch]
Harmonicity forces the field evolution to be canonical and syntax-independent.
Quine--stability forces the hypergraph to be in unique normal form.
Critical sets of harmonic fields correspond bijectively to strata of quine--stable graphs.
Thus morphisms correspond to curvature-/entropy-preserving maps,
which are identical on both sides.
\end{proof}

\section{Essential Surjectivity on Stable Objects}

\begin{theorem}[Essential Surjectivity]
Every quine--stable $\mathcal{G}$ arises (up to isomorphism)
as $\mathbb{R}(\Phi,\mathbf{v},S)$ for some harmonic RSVP field configuration.
\end{theorem}

\begin{proof}
Given quine--stable $\mathcal{G}$, construct the harmonic triple
$(\Phi_{\mathcal{G}},\mathbf{v}_{\mathcal{G}},S_{\mathcal{G}})$ as above.
The embedding constraints guarantee that $\mathcal{G}$
is the full critical skeleton of the resulting RSVP configuration.
\end{proof}

\section{Preservation and Reflection of Modal Operators}

\begin{theorem}[Modal Preservation and Reflection]
\[
\mathbb{R}(\Box_{\mathrm{RSVP}}P) = \Box_{\mathrm{Polyx}}\mathbb{R}(P),
\qquad
\mathbb{R}^{-1}(\Box_{\mathrm{Polyx}}Q) = \Box_{\mathrm{RSVP}}\mathbb{R}^{-1}(Q).
\]
\end{theorem}

\begin{proof}
Both $\Box$ modalities are inductively defined via stability under evolutions
(harmonic flow for RSVP, EHR steps for Polyx).
But $\mathbb{R}$ identifies flows with rewrite sequences
and rewrite sequences with flows (Ch.~33).
Hence preservation and reflection follow definitionally.
\end{proof}

\section{Duality Isomorphism of Moduli Spaces}

Let
\[
\mathcal{M}_{\mathrm{gal}} := \pi_0(\text{harmonic RSVP fields}/\!\!\sim),
\qquad
\mathcal{M}_{\mathrm{quine}} := \pi_0(\text{quine--stable Polyx graphs}/\!\!\sim),
\]
be the homotopy moduli spaces.

\begin{theorem}[Duality Isomorphism Theorem]
\[
\mathbb{R} : \mathcal{M}_{\mathrm{gal}} \;\xrightarrow{\;\cong\;}\;
\mathcal{M}_{\mathrm{quine}}.
\]
\end{theorem}

\begin{proof}
Full faithfulness and essential surjectivity on stable objects
imply categorical equivalence.
Passing to connected components yields the homotopy equivalence.
\end{proof}

\section{The Universal Media Quine as the Image of the Conformal Vacuum}

\begin{theorem}
\[
\mathbb{R}(\text{conformal vacuum}) = \mathfrak{U},
\qquad
\mathbb{R}^{-1}(\mathfrak{U}) = \text{conformal vacuum}.
\]
\end{theorem}

\begin{proof}
The conformal vacuum has zero curvature, zero entropy gradient, and a single critical stratum.
Thus its image under $\mathbb{R}$ is precisely the unique quine--stable hypergraph
with vanishing curvature and trivial tag.
Conversely, applying $\mathbb{R}^{-1}$ recovers the unique flat RSVP configuration.
\end{proof}
