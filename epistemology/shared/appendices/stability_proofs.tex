\section{Stability Proofs}

This appendix establishes robustness properties of agentic systems.

\subsection{Stability Under Perturbation}

\begin{definition}
A system is semantically stable if small perturbations to inputs do not induce execution when refusal is required.
\end{definition}

\begin{theorem}[Refusal Stability]
Agents are stable under perturbation; didactic interfaces are not.
\end{theorem}

\begin{proof}
For agents, constraint evaluation precedes execution, absorbing perturbations via refusal.  
For didactic interfaces, execution is decoupled from constraints; perturbations propagate unchecked.
\end{proof}

\subsection{Goodhart Instability}

\begin{theorem}
Any fixed optimization target induces semantic instability under repeated optimization.
\end{theorem}

\begin{proof}
Optimization collapses equivalence classes defined by the target. Once stabilized, semantic diversity is eliminated, inducing Goodhart failure.
\end{proof}

\subsection{Invariant Alignment}

\begin{corollary}
Alignment is stable iff misaligned states are unreachable in $\States$.
\end{corollary}
