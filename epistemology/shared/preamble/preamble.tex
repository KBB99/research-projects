% ============================================================
% Spherepop / RSVP Shared LaTeX Preamble
% Author: Flyxion
% Purpose: Multi-journal, theory-first, non-didactic manuscripts
% ============================================================

% -------------------------------
% Engine and Encoding
% -------------------------------
\usepackage[T1]{fontenc}
\usepackage[utf8]{inputenc}

% -------------------------------
% Page Layout
% -------------------------------
\usepackage[a4paper,margin=1in]{geometry}
\usepackage{setspace}
\setstretch{1.15}

% -------------------------------
% Fonts (Conservative, Journal-Safe)
% -------------------------------
\usepackage{lmodern}        % Clean serif + math
\usepackage{microtype}     % Subtle typographic improvements

% -------------------------------
% Mathematics
% -------------------------------
\usepackage{amsmath}
\usepackage{amssymb}
\usepackage{amsfonts}
\usepackage{amsthm}
\usepackage{mathtools}

% Do NOT overdefine math macros here.
% All theory-specific macros belong in shared/macros/macros.tex

% -------------------------------
% Theorem Environments
% -------------------------------
\theoremstyle{plain}
\newtheorem{theorem}{Theorem}[section]
\newtheorem{lemma}[theorem]{Lemma}
\newtheorem{proposition}[theorem]{Proposition}
\newtheorem{corollary}[theorem]{Corollary}

\theoremstyle{definition}
\newtheorem{definition}[theorem]{Definition}

\theoremstyle{remark}
\newtheorem{remark}[theorem]{Remark}
\newtheorem{example}[theorem]{Example}

% -------------------------------
% Citations and Bibliography
% -------------------------------
\usepackage[
  backend=biber,
  style=authoryear,
  maxcitenames=2,
  maxbibnames=99,
  doi=false,
  isbn=false,
  url=false
]{biblatex}

% The bibliography file is always shared
\addbibresource{shared/bib/references.bib}

% -------------------------------
% Hyperlinks (Muted, Non-Distracting)
% -------------------------------
\usepackage[hidelinks]{hyperref}

% -------------------------------
% Figures and Tables (Minimal)
% -------------------------------
\usepackage{graphicx}
\usepackage{booktabs}

% -------------------------------
% Lists
% -------------------------------
\usepackage{enumitem}
\setlist{nosep}

% -------------------------------
% Section Styling (No Visual Drama)
% -------------------------------
\usepackage{titlesec}

\titleformat{\section}
  {\normalfont\Large\bfseries}
  {\thesection}{1em}{}

\titleformat{\subsection}
  {\normalfont\large\bfseries}
  {\thesubsection}{1em}{}

\titleformat{\subsubsection}
  {\normalfont\normalsize\bfseries}
  {\thesubsubsection}{1em}{}

% -------------------------------
% Footnotes
% -------------------------------
\usepackage[bottom]{footmisc}

% -------------------------------
% Quotation Handling
% -------------------------------
\usepackage{csquotes}

% -------------------------------
% Philosophy-Friendly Tweaks
% -------------------------------

% Avoid widow/orphan lines
\clubpenalty=10000
\widowpenalty=10000

% Prevent overfull boxes from breaking layout
\emergencystretch=3em

% -------------------------------
% Intentional Omissions (By Design)
% -------------------------------
% - No algorithm environments
% - No code listings
% - No color
% - No diagram libraries
% - No tutorial-style boxes
%
% These omissions are structural, not aesthetic.
% They prevent accidental didacticization.

% ============================================================
% End of shared preamble
% ============================================================
